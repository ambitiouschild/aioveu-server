#### 2. Service层

```
@Service
@Slf4j
public class WxAuthService {
    
    @Autowired
    private TenantWxConfigService wxConfigService;
    
    /**
     * 微信登录验证
     * @param tenantCode 租户编码
     * @param code 微信code
     * @return 用户信息
     */
    public WxAuthResult wxLogin(String tenantCode, String code) {
        // 1. 根据租户编码查询微信配置
        TenantWxConfig wxConfig = wxConfigService.getConfigByTenantCode(tenantCode);
        if (wxConfig == null || wxConfig.getStatus() != 1) {
            throw new BusinessException("租户未配置微信登录");
        }
        
        // 2. 调用微信接口获取openid
        WxSessionInfo sessionInfo = getWxSessionInfo(
            wxConfig.getAppId(), 
            wxConfig.getAppSecret(), 
            code
        );
        
        // 3. 查询或创建用户
        User user = findOrCreateUser(tenantCode, sessionInfo.getOpenid());
        
        // 4. 生成JWT token
        String token = generateToken(user, tenantCode);
        
        return WxAuthResult.builder()
                .token(token)
                .openid(sessionInfo.getOpenid())
                .unionid(sessionInfo.getUnionid())
                .userInfo(user)
                .build();
    }
    
    /**
     * 调用微信接口获取session信息
     */
    private WxSessionInfo getWxSessionInfo(String appId, String appSecret, String code) {
        String url = String.format(
            "https://api.weixin.qq.com/sns/jscode2session?appid=%s&secret=%s&js_code=%s&grant_type=authorization_code",
            appId, appSecret, code
        );
        
        try {
            RestTemplate restTemplate = new RestTemplate();
            String response = restTemplate.getForObject(url, String.class);
            JSONObject json = new JSONObject(response);
            
            if (json.has("errcode")) {
                log.error("微信登录失败: appId={}, error={}", appId, json.toString());
                throw new BusinessException("微信登录失败: " + json.getString("errmsg"));
            }
            
            return WxSessionInfo.builder()
                    .openid(json.getString("openid"))
                    .sessionKey(json.getString("session_key"))
                    .unionid(json.optString("unionid"))
                    .build();
                    
        } catch (Exception e) {
            log.error("调用微信接口失败: appId={}", appId, e);
            throw new BusinessException("微信服务异常");
        }
    }
    
    /**
     * 查询或创建用户
     */
    private User findOrCreateUser(String tenantCode, String openid) {
        // 1. 查询用户是否存在
        User user = userService.findByTenantAndOpenid(tenantCode, openid);
        
        // 2. 如果不存在，创建新用户
        if (user == null) {
            user = new User();
            user.setTenantCode(tenantCode);
            user.setOpenid(openid);
            user.setUsername("wx_" + openid.substring(0, 8));  // 生成默认用户名
            user.setStatus(1);
            userService.save(user);
        }
        
        return user;
    }
}
```

#### 3. Controller层

```
@RestController
@RequestMapping("/api/auth/wx")
@Slf4j
public class WxAuthController {
    
    @Autowired
    private WxAuthService wxAuthService;
    
    /**
     * 微信登录接口
     * @param tenantCode 租户编码（从前端传递）
     * @param code 微信code
     */
    @PostMapping("/login")
    public Result<WxAuthResult> wxLogin(
            @RequestHeader("X-Tenant-Code") String tenantCode,
            @RequestParam String code) {
        
        log.info("微信登录请求: tenantCode={}, codeLength={}", 
                tenantCode, code.length());
        
        WxAuthResult result = wxAuthService.wxLogin(tenantCode, code);
        return Result.success(result);
    }
    
    /**
     * 获取租户的微信配置（用于前端初始化）
     */
    @GetMapping("/config")
    public Result<Map<String, String>> getWxConfig(
            @RequestHeader("X-Tenant-Code") String tenantCode) {
        
        TenantWxConfig config = wxConfigService.getConfigByTenantCode(tenantCode);
        if (config == null) {
            return Result.error("租户未配置微信登录");
        }
        
        Map<String, String> configMap = new HashMap<>();
        configMap.put("appId", config.getAppId());
        // 注意：appSecret不能返回给前端！
        configMap.put("appName", config.getAppName());
        configMap.put("appType", config.getAppType());
        
        return Result.success(configMap);
    }
}
```

#### 4. 租户过滤器

```
@Component
@Slf4j
public class TenantFilter extends OncePerRequestFilter {
    
    @Override
    protected void doFilterInternal(HttpServletRequest request, 
                                   HttpServletResponse response, 
                                   FilterChain filterChain) 
            throws ServletException, IOException {
        
        // 1. 从请求头获取租户编码
        String tenantCode = request.getHeader("X-Tenant-Code");
        
        // 2. 如果没有租户编码，尝试从子域名获取
        if (StringUtils.isBlank(tenantCode)) {
            tenantCode = extractTenantFromDomain(request);
        }
        
        // 3. 验证租户是否存在且启用
        if (StringUtils.isNotBlank(tenantCode)) {
            Tenant tenant = tenantService.getByCode(tenantCode);
            if (tenant == null || tenant.getStatus() != 1) {
                response.setStatus(HttpStatus.FORBIDDEN.value());
                response.getWriter().write("租户不存在或已禁用");
                return;
            }
            
            // 4. 设置租户上下文
            TenantContext.setCurrentTenant(tenant);
        }
        
        try {
            filterChain.doFilter(request, response);
        } finally {
            // 清理上下文
            TenantContext.clear();
        }
    }
}
```

#### 5. 租户上下文

```
public class TenantContext {
    private static final ThreadLocal<Tenant> currentTenant = new ThreadLocal<>();
    
    public static void setCurrentTenant(Tenant tenant) {
        currentTenant.set(tenant);
    }
    
    public static Tenant getCurrentTenant() {
        return currentTenant.get();
    }
    
    public static String getCurrentTenantCode() {
        Tenant tenant = currentTenant.get();
        return tenant != null ? tenant.getTenantCode() : null;
    }
    
    public static void clear() {
        currentTenant.remove();
    }
}
```

### 前端调用示例

```
// 前端需要知道当前是哪个租户
const tenantCode = 'tenant001';  // 可以从URL参数、localStorage等获取

// 1. 调用登录接口
async function wxLogin(code) {
    const response = await fetch('/api/auth/wx/login', {
        method: 'POST',
        headers: {
            'Content-Type': 'application/x-www-form-urlencoded',
            'X-Tenant-Code': tenantCode
        },
        body: `code=${code}`
    });
    
    const result = await response.json();
    if (result.code === 0) {
        // 登录成功，保存token
        localStorage.setItem('token', result.data.token);
        localStorage.setItem('tenantCode', tenantCode);
    }
    return result;
}

// 2. 微信小程序端获取code
// 注意：不同租户的小程序appid不同
wx.login({
  success: async (res) => {
    if (res.code) {
      const loginResult = await wxLogin(res.code);
      console.log('登录结果:', loginResult);
    }
  }
});
```

### 管理后台配置界面

```
@RestController
@RequestMapping("/api/admin/wx-config")
public class WxConfigAdminController {
    
    /**
     * 为租户配置微信参数
     */
    @PostMapping("/configure")
    public Result<Void> configureWx(@RequestBody WxConfigRequest request) {
        // 验证参数
        if (StringUtils.isBlank(request.getAppId()) || 
            StringUtils.isBlank(request.getAppSecret())) {
            return Result.error("appId和appSecret不能为空");
        }
        
        // 测试配置是否有效
        boolean valid = testWxConfig(request.getAppId(), request.getAppSecret());
        if (!valid) {
            return Result.error("微信配置验证失败，请检查appId和appSecret");
        }
        
        // 保存配置
        wxConfigService.saveOrUpdateConfig(
            request.getTenantId(),
            request.getAppType(),
            request.getAppId(),
            request.getAppSecret(),
            request.getAppName()
        );
        
        return Result.success();
    }
    
    /**
     * 测试微信配置
     */
    private boolean testWxConfig(String appId, String appSecret) {
        try {
            // 使用一个测试code（可以是任意字符串）调用微信接口
            // 如果返回错误码不是40029（无效code），说明appId/appSecret有效
            String url = String.format(
                "https://api.weixin.qq.com/sns/jscode2session?appid=%s&secret=%s&js_code=test_code&grant_type=authorization_code",
                appId, appSecret
            );
            
            RestTemplate restTemplate = new RestTemplate();
            String response = restTemplate.getForObject(url, String.class);
            JSONObject json = new JSONObject(response);
            
            // 40029是无效code，其他错误说明配置有问题
            return json.optInt("errcode", 0) == 40029;
        } catch (Exception e) {
            return false;
        }
    }
}
```

### 总结

**多租户微信登录的核心流程：**

1. **前端传递租户标识**（通过header、子域名、URL参数等）
2. **后端根据租户查询对应的appid/appsecret**
3. **使用租户专属的微信配置调用微信接口**
4. **生成租户内的用户账号和token**

**优点：**

- 各租户完全隔离
- 可以独立管理自己的微信配置
- 灵活支持不同的小程序/公众号
- 易于扩展和维护

这样每个租户都可以有自己的小程序，用户数据完全隔离，符合多租户架构的设计原则。
