理解了！您的切换租户接口需要根据不同的授权模式生成对应格式的新令牌。以下是完整的实现方案：

## 1. 创建统一的租户切换Token服务

```
package com.aioveu.auth.service.tenant;

import com.aioveu.auth.domain.SysUserDetails;
import com.aioveu.auth.service.TenantService;
import lombok.RequiredArgsConstructor;
import lombok.extern.slf4j.Slf4j;
import org.springframework.security.authentication.UsernamePasswordAuthenticationToken;
import org.springframework.security.core.Authentication;
import org.springframework.security.core.context.SecurityContextHolder;
import org.springframework.security.oauth2.core.*;
import org.springframework.security.oauth2.core.endpoint.OAuth2ParameterNames;
import org.springframework.security.oauth2.core.oidc.OidcIdToken;
import org.springframework.security.oauth2.core.oidc.OidcScopes;
import org.springframework.security.oauth2.core.oidc.endpoint.OidcParameterNames;
import org.springframework.security.oauth2.jwt.Jwt;
import org.springframework.security.oauth2.server.authorization.OAuth2Authorization;
import org.springframework.security.oauth2.server.authorization.OAuth2AuthorizationService;
import org.springframework.security.oauth2.server.authorization.OAuth2TokenType;
import org.springframework.security.oauth2.server.authorization.authentication.OAuth2ClientAuthenticationToken;
import org.springframework.security.oauth2.server.authorization.client.RegisteredClient;
import org.springframework.security.oauth2.server.authorization.client.RegisteredClientRepository;
import org.springframework.security.oauth2.server.authorization.context.AuthorizationServerContextHolder;
import org.springframework.security.oauth2.server.authorization.token.DefaultOAuth2TokenContext;
import org.springframework.security.oauth2.server.authorization.token.OAuth2TokenContext;
import org.springframework.security.oauth2.server.authorization.token.OAuth2TokenGenerator;
import org.springframework.stereotype.Service;
import org.springframework.util.StringUtils;

import java.util.*;

/**
 * 租户切换Token服务
 * 根据原令牌的授权模式，生成包含新租户ID的新令牌
 */
@Slf4j
@Service
@RequiredArgsConstructor
public class TenantSwitchTokenService {
    
    private final OAuth2TokenGenerator<? extends OAuth2Token> tokenGenerator;
    private final OAuth2AuthorizationService authorizationService;
    private final RegisteredClientRepository registeredClientRepository;
    private final TenantService tenantService;
    
    /**
     * 切换租户并生成新令牌
     */
    public OAuth2AccessToken switchTenant(Long newTenantId, String currentAccessTokenValue) {
        log.info("开始切换租户: tenantId={}", newTenantId);
        
        // 1. 获取当前认证和授权信息
        Authentication currentAuth = SecurityContextHolder.getContext().getAuthentication();
        if (!(currentAuth.getPrincipal() instanceof SysUserDetails currentDetails)) {
            throw new OAuth2AuthenticationException("无效的用户认证");
        }
        
        // 2. 验证租户访问权限
        if (!tenantService.canAccessTenant(currentDetails.getUserId(), newTenantId)) {
            throw new OAuth2AuthenticationException("无权限访问该租户");
        }
        
        // 3. 获取当前授权信息
        OAuth2Authorization currentAuthorization = authorizationService.findByToken(
            currentAccessTokenValue, 
            OAuth2TokenType.ACCESS_TOKEN
        );
        
        if (currentAuthorization == null) {
            throw new OAuth2AuthenticationException("原令牌无效");
        }
        
        // 4. 根据原授权模式生成新令牌
        return generateNewTokenByGrantType(newTenantId, currentDetails, currentAuthorization);
    }
    
    /**
     * 根据原授权模式生成新令牌
     */
    private OAuth2AccessToken generateNewTokenByGrantType(
            Long newTenantId, 
            SysUserDetails currentDetails,
            OAuth2Authorization currentAuthorization) {
        
        AuthorizationGrantType grantType = currentAuthorization.getAuthorizationGrantType();
        log.info("原令牌授权模式: {}", grantType.getValue());
        
        switch (grantType.getValue()) {
            case "password":
                return generatePasswordModeToken(newTenantId, currentDetails, currentAuthorization);
            case "authorization_code":
                return generateAuthCodeModeToken(newTenantId, currentDetails, currentAuthorization);
            case "client_credentials":
                return generateClientCredentialsToken(newTenantId, currentDetails, currentAuthorization);
            case "refresh_token":
                return generateRefreshTokenModeToken(newTenantId, currentDetails, currentAuthorization);
            default:
                return generateGenericToken(newTenantId, currentDetails, currentAuthorization);
        }
    }
    
    /**
     * 密码模式令牌生成
     */
    private OAuth2AccessToken generatePasswordModeToken(
            Long newTenantId, 
            SysUserDetails currentDetails,
            OAuth2Authorization currentAuthorization) {
        
        RegisteredClient registeredClient = getRegisteredClient(currentAuthorization);
        Set<String> authorizedScopes = currentAuthorization.getAuthorizedScopes();
        
        // 创建包含新租户ID的用户认证
        SysUserDetails newDetails = createNewUserDetails(currentDetails, newTenantId);
        Authentication newPrincipal = new UsernamePasswordAuthenticationToken(
            newDetails,
            currentDetails.getPassword(),
            currentDetails.getAuthorities()
        );
        
        // 构建令牌上下文
        DefaultOAuth2TokenContext.Builder contextBuilder = DefaultOAuth2TokenContext.builder()
            .registeredClient(registeredClient)
            .principal(newPrincipal)
            .authorizationServerContext(AuthorizationServerContextHolder.getContext())
            .authorizedScopes(authorizedScopes)
            .authorizationGrantType(AuthorizationGrantType.PASSWORD)
            .put("tenant_id", newTenantId);
        
        // 生成访问令牌
        OAuth2TokenContext tokenContext = contextBuilder.tokenType(OAuth2TokenType.ACCESS_TOKEN).build();
        OAuth2Token generatedAccessToken = tokenGenerator.generate(tokenContext);
        
        if (!(generatedAccessToken instanceof OAuth2AccessToken newAccessToken)) {
            throw new OAuth2AuthenticationException("访问令牌生成失败");
        }
        
        // 生成刷新令牌
        OAuth2RefreshToken refreshToken = generateRefreshTokenIfNeeded(
            contextBuilder, registeredClient, currentAuthorization);
        
        // 生成ID令牌（如果原令牌有）
        OidcIdToken idToken = generateIdTokenIfNeeded(
            contextBuilder, authorizedScopes, currentAuthorization, newDetails);
        
        // 保存新授权记录
        saveNewAuthorization(currentAuthorization, newAccessToken, refreshToken, idToken, newPrincipal);
        
        return newAccessToken;
    }
    
    /**
     * 授权码模式令牌生成
     */
    private OAuth2AccessToken generateAuthCodeModeToken(
            Long newTenantId, 
            SysUserDetails currentDetails,
            OAuth2Authorization currentAuthorization) {
        
        RegisteredClient registeredClient = getRegisteredClient(currentAuthorization);
        Set<String> authorizedScopes = currentAuthorization.getAuthorizedScopes();
        
        // 模拟授权码模式的上下文
        SysUserDetails newDetails = createNewUserDetails(currentDetails, newTenantId);
        Authentication newPrincipal = new UsernamePasswordAuthenticationToken(
            newDetails,
            null, // 授权码模式没有密码
            currentDetails.getAuthorities()
        );
        
        // 创建授权码认证令牌
        OAuth2ClientAuthenticationToken clientAuth = new OAuth2ClientAuthenticationToken(
            registeredClient,
            ClientAuthenticationMethod.CLIENT_SECRET_BASIC,
            registeredClient.getClientSecret()
        );
        
        DefaultOAuth2TokenContext.Builder contextBuilder = DefaultOAuth2TokenContext.builder()
            .registeredClient(registeredClient)
            .principal(newPrincipal)
            .authorizationServerContext(AuthorizationServerContextHolder.getContext())
            .authorizedScopes(authorizedScopes)
            .authorizationGrantType(AuthorizationGrantType.AUTHORIZATION_CODE)
            .put("tenant_id", newTenantId)
            .put(OAuth2ParameterNames.CODE, "switch_tenant_code"); // 模拟授权码
        
        // 生成令牌
        OAuth2TokenContext tokenContext = contextBuilder.tokenType(OAuth2TokenType.ACCESS_TOKEN).build();
        OAuth2Token generatedAccessToken = tokenGenerator.generate(tokenContext);
        
        if (!(generatedAccessToken instanceof OAuth2AccessToken newAccessToken)) {
            throw new OAuth2AuthenticationException("访问令牌生成失败");
        }
        
        saveNewAuthorization(currentAuthorization, newAccessToken, null, null, newPrincipal);
        return newAccessToken;
    }
    
    /**
     * 客户端凭证模式令牌生成
     */
    private OAuth2AccessToken generateClientCredentialsToken(
            Long newTenantId, 
            SysUserDetails currentDetails,
            OAuth2Authorization currentAuthorization) {
        
        RegisteredClient registeredClient = getRegisteredClient(currentAuthorization);
        Set<String> authorizedScopes = currentAuthorization.getAuthorizedScopes();
        
        // 客户端凭证模式通常没有用户信息，但我们需要添加租户信息
        DefaultOAuth2TokenContext.Builder contextBuilder = DefaultOAuth2TokenContext.builder()
            .registeredClient(registeredClient)
            .principal(currentDetails) // 保持原用户信息
            .authorizationServerContext(AuthorizationServerContextHolder.getContext())
            .authorizedScopes(authorizedScopes)
            .authorizationGrantType(AuthorizationGrantType.CLIENT_CREDENTIALS)
            .put("tenant_id", newTenantId);
        
        OAuth2TokenContext tokenContext = contextBuilder.tokenType(OAuth2TokenType.ACCESS_TOKEN).build();
        OAuth2Token generatedAccessToken = tokenGenerator.generate(tokenContext);
        
        if (!(generatedAccessToken instanceof OAuth2AccessToken newAccessToken)) {
            throw new OAuth2AuthenticationException("访问令牌生成失败");
        }
        
        // 更新原授权记录（客户端凭证模式通常不创建新记录）
        updateAuthorizationWithNewTenant(currentAuthorization, newAccessToken, newTenantId);
        return newAccessToken;
    }
    
    /**
     * 刷新令牌模式
     */
    private OAuth2AccessToken generateRefreshTokenModeToken(
            Long newTenantId, 
            SysUserDetails currentDetails,
            OAuth2Authorization currentAuthorization) {
        
        RegisteredClient registeredClient = getRegisteredClient(currentAuthorization);
        Set<String> authorizedScopes = currentAuthorization.getAuthorizedScopes();
        
        // 使用刷新令牌生成新访问令牌
        SysUserDetails newDetails = createNewUserDetails(currentDetails, newTenantId);
        Authentication newPrincipal = new UsernamePasswordAuthenticationToken(
            newDetails,
            currentDetails.getPassword(),
            currentDetails.getAuthorities()
        );
        
        DefaultOAuth2TokenContext.Builder contextBuilder = DefaultOAuth2TokenContext.builder()
            .registeredClient(registeredClient)
            .principal(newPrincipal)
            .authorizationServerContext(AuthorizationServerContextHolder.getContext())
            .authorizedScopes(authorizedScopes)
            .authorizationGrantType(AuthorizationGrantType.REFRESH_TOKEN)
            .put("tenant_id", newTenantId)
            .put(OAuth2ParameterNames.REFRESH_TOKEN, 
                currentAuthorization.getRefreshToken().getToken().getTokenValue());
        
        OAuth2TokenContext tokenContext = contextBuilder.tokenType(OAuth2TokenType.ACCESS_TOKEN).build();
        OAuth2Token generatedAccessToken = tokenGenerator.generate(tokenContext);
        
        if (!(generatedAccessToken instanceof OAuth2AccessToken newAccessToken)) {
            throw new OAuth2AuthenticationException("访问令牌生成失败");
        }
        
        // 生成新的刷新令牌
        OAuth2RefreshToken newRefreshToken = generateRefreshToken(contextBuilder, registeredClient);
        
        saveNewAuthorization(currentAuthorization, newAccessToken, newRefreshToken, null, newPrincipal);
        return newAccessToken;
    }
    
    /**
     * 通用令牌生成
     */
    private OAuth2AccessToken generateGenericToken(
            Long newTenantId, 
            SysUserDetails currentDetails,
            OAuth2Authorization currentAuthorization) {
        
        RegisteredClient registeredClient = getRegisteredClient(currentAuthorization);
        Set<String> authorizedScopes = currentAuthorization.getAuthorizedScopes();
        
        SysUserDetails newDetails = createNewUserDetails(currentDetails, newTenantId);
        Authentication newPrincipal = new UsernamePasswordAuthenticationToken(
            newDetails,
            currentDetails.getPassword(),
            currentDetails.getAuthorities()
        );
        
        DefaultOAuth2TokenContext.Builder contextBuilder = DefaultOAuth2TokenContext.builder()
            .registeredClient(registeredClient)
            .principal(newPrincipal)
            .authorizationServerContext(AuthorizationServerContextHolder.getContext())
            .authorizedScopes(authorizedScopes)
            .authorizationGrantType(currentAuthorization.getAuthorizationGrantType())
            .put("tenant_id", newTenantId);
        
        OAuth2TokenContext tokenContext = contextBuilder.tokenType(OAuth2TokenType.ACCESS_TOKEN).build();
        OAuth2Token generatedAccessToken = tokenGenerator.generate(tokenContext);
        
        if (!(generatedAccessToken instanceof OAuth2AccessToken newAccessToken)) {
            throw new OAuth2AuthenticationException("访问令牌生成失败");
        }
        
        saveNewAuthorization(currentAuthorization, newAccessToken, null, null, newPrincipal);
        return newAccessToken;
    }
    
    // ========== 辅助方法 ==========
    
    private SysUserDetails createNewUserDetails(SysUserDetails original, Long newTenantId) {
        // 深拷贝并更新租户ID
        return SysUserDetails.builder()
            .userId(original.getUserId())
            .username(original.getUsername())
            .tenantId(newTenantId)
            .deptId(original.getDeptId())
            .dataScope(original.getDataScope())
            .canSwitchTenant(original.getCanSwitchTenant())
            .authorities(original.getAuthorities())
            .password(original.getPassword())
            .accountNonExpired(original.isAccountNonExpired())
            .accountNonLocked(original.isAccountNonLocked())
            .credentialsNonExpired(original.isCredentialsNonExpired())
            .enabled(original.isEnabled())
            .build();
    }
    
    private RegisteredClient getRegisteredClient(OAuth2Authorization authorization) {
        String clientId = authorization.getRegisteredClientId();
        RegisteredClient client = registeredClientRepository.findByClientId(clientId);
        if (client == null) {
            throw new OAuth2AuthenticationException("客户端不存在");
        }
        return client;
    }
    
    private OAuth2RefreshToken generateRefreshTokenIfNeeded(
            DefaultOAuth2TokenContext.Builder contextBuilder,
            RegisteredClient registeredClient,
            OAuth2Authorization currentAuthorization) {
        
        if (registeredClient.getAuthorizationGrantTypes().contains(AuthorizationGrantType.REFRESH_TOKEN)) {
            OAuth2TokenContext refreshTokenContext = contextBuilder
                .tokenType(OAuth2TokenType.REFRESH_TOKEN)
                .build();
            
            OAuth2Token generatedRefreshToken = tokenGenerator.generate(refreshTokenContext);
            if (generatedRefreshToken instanceof OAuth2RefreshToken) {
                return (OAuth2RefreshToken) generatedRefreshToken;
            }
        }
        return null;
    }
    
    private OAuth2RefreshToken generateRefreshToken(
            DefaultOAuth2TokenContext.Builder contextBuilder,
            RegisteredClient registeredClient) {
        
        OAuth2TokenContext refreshTokenContext = contextBuilder
            .tokenType(OAuth2TokenType.REFRESH_TOKEN)
            .build();
        
        OAuth2Token generatedRefreshToken = tokenGenerator.generate(refreshTokenContext);
        if (generatedRefreshToken instanceof OAuth2RefreshToken) {
            return (OAuth2RefreshToken) generatedRefreshToken;
        }
        return null;
    }
    
    private OidcIdToken generateIdTokenIfNeeded(
            DefaultOAuth2TokenContext.Builder contextBuilder,
            Set<String> authorizedScopes,
            OAuth2Authorization currentAuthorization,
            SysUserDetails newDetails) {
        
        if (authorizedScopes.contains(OidcScopes.OPENID)) {
            OAuth2TokenContext idTokenContext = contextBuilder
                .tokenType(new OAuth2TokenType(OidcParameterNames.ID_TOKEN))
                .build();
            
            OAuth2Token generatedIdToken = tokenGenerator.generate(idTokenContext);
            if (generatedIdToken instanceof Jwt) {
                Jwt jwt = (Jwt) generatedIdToken;
                return new OidcIdToken(
                    jwt.getTokenValue(),
                    jwt.getIssuedAt(),
                    jwt.getExpiresAt(),
                    jwt.getClaims()
                );
            }
        }
        return null;
    }
    
    private void saveNewAuthorization(
            OAuth2Authorization currentAuthorization,
            OAuth2AccessToken newAccessToken,
            OAuth2RefreshToken refreshToken,
            OidcIdToken idToken,
            Authentication newPrincipal) {
        
        OAuth2Authorization.Builder builder = OAuth2Authorization.from(currentAuthorization)
            .principalName(newPrincipal.getName())
            .attribute(Principal.class.getName(), newPrincipal)
            .token(newAccessToken, metadata -> {
                if (newAccessToken instanceof ClaimAccessor) {
                    metadata.put(OAuth2Authorization.Token.CLAIMS_METADATA_NAME, 
                        ((ClaimAccessor) newAccessToken).getClaims());
                }
            });
        
        if (refreshToken != null) {
            builder.refreshToken(refreshToken);
        }
        
        if (idToken != null) {
            builder.token(idToken, metadata -> 
                metadata.put(OAuth2Authorization.Token.CLAIMS_METADATA_NAME, idToken.getClaims()));
        }
        
        OAuth2Authorization newAuthorization = builder.build();
        authorizationService.save(newAuthorization);
        
        // 可选：移除旧授权记录
        authorizationService.remove(currentAuthorization);
    }
    
    private void updateAuthorizationWithNewTenant(
            OAuth2Authorization authorization,
            OAuth2AccessToken newAccessToken,
            Long newTenantId) {
        
        Authentication currentPrincipal = authorization.getAttribute(Principal.class.getName());
        if (currentPrincipal.getPrincipal() instanceof SysUserDetails currentDetails) {
            SysUserDetails newDetails = createNewUserDetails(currentDetails, newTenantId);
            Authentication newPrincipal = new UsernamePasswordAuthenticationToken(
                newDetails,
                currentPrincipal.getCredentials(),
                currentPrincipal.getAuthorities()
            );
            
            OAuth2Authorization updatedAuth = OAuth2Authorization.from(authorization)
                .principalName(newPrincipal.getName())
                .attribute(Principal.class.getName(), newPrincipal)
                .token(newAccessToken, metadata -> {
                    if (newAccessToken instanceof ClaimAccessor) {
                        metadata.put(OAuth2Authorization.Token.CLAIMS_METADATA_NAME, 
                            ((ClaimAccessor) newAccessToken).getClaims());
                    }
                })
                .build();
            
            authorizationService.save(updatedAuth);
        }
    }
}
```

## 2. 更新切换租户接口

```
@Operation(summary = "切换租户")
@PostMapping("/switch-tenant")
public Result<OAuth2AccessToken> switchTenant(@RequestParam Long tenantId, 
                                             HttpServletRequest request) {
    try {
        // 1. 验证权限
        Authentication authentication = SecurityContextHolder.getContext().getAuthentication();
        if (authentication == null || !(authentication.getPrincipal() instanceof SysUserDetails details)) {
            return Result.failed(ResultCode.ACCESS_TOKEN_INVALID);
        }
        
        if (!details.getCanSwitchTenant()) {
            return Result.failed("无租户切换权限");
        }
        
        if (!tenantService.canAccessTenant(details.getUserId(), tenantId)) {
            return Result.failed("无权限访问该租户");
        }
        
        // 2. 获取当前访问令牌
        String currentToken = extractAccessToken(request);
        if (!StringUtils.hasText(currentToken)) {
            return Result.failed("无法获取当前令牌");
        }
        
        // 3. 生成包含新租户ID的令牌
        OAuth2AccessToken newToken = tenantSwitchTokenService.switchTenant(tenantId, currentToken);
        
        // 4. 返回新令牌
        return Result.success(newToken);
        
    } catch (OAuth2AuthenticationException e) {
        log.error("租户切换失败", e);
        return Result.failed(e.getError().getDescription());
    } catch (Exception e) {
        log.error("租户切换异常", e);
        return Result.failed("系统异常");
    }
}

private String extractAccessToken(HttpServletRequest request) {
    String authHeader = request.getHeader("Authorization");
    if (StringUtils.hasText(authHeader) && authHeader.startsWith("Bearer ")) {
        return authHeader.substring(7);
    }
    return null;
}
```

## 3. 创建自定义的Token定制器（可选）

如果需要在新令牌中包含租户信息：

```
@Component
public class TenantAwareTokenCustomizer implements OAuth2TokenCustomizer<OAuth2TokenClaimsContext> {
    
    @Override
    public void customize(OAuth2TokenClaimsContext context) {
        OAuth2TokenContext tokenContext = context.getTokenContext();
        
        // 从上下文获取租户ID
        Long tenantId = tokenContext.get("tenant_id");
        if (tenantId != null) {
            context.getClaims().claim("tenant_id", tenantId);
            
            // 如果用户详情中有租户ID，也添加进去
            if (tokenContext.getPrincipal().getPrincipal() instanceof SysUserDetails userDetails) {
                context.getClaims().claim("tenant_id", userDetails.getTenantId());
            }
        }
    }
}
```

## 关键特性：

1. **多模式支持**：根据原令牌的授权模式（password、authorization_code、client_credentials、refresh_token）生成对应格式的新令牌
2. **租户信息传递**：在令牌上下文中添加租户ID，确保自定义Token定制器能获取到
3. **授权记录更新**：保存新的授权记录，使旧令牌失效
4. **令牌声明**：在新令牌的声明中包含租户ID
5. **安全性**：验证用户权限，防止越权切换

这样您的切换租户接口就能根据不同的OAuth2授权模式生成正确格式的新令牌了。
