```
package com.aioveu.auth.controller;

import io.swagger.v3.oas.annotations.Operation;

import io.swagger.v3.oas.annotations.tags.Tag;
import lombok.RequiredArgsConstructor;
import lombok.extern.slf4j.Slf4j;
import org.springdoc.core.annotations.ParameterObject;
import org.springframework.data.domain.Page;
import org.springframework.data.domain.Pageable;
import org.springframework.http.HttpStatus;
import org.springframework.http.ResponseEntity;
import org.springframework.security.access.prepost.PreAuthorize;
import org.springframework.web.bind.annotation.*;

import javax.validation.Valid;
import java.net.URI;
import java.util.List;

@Slf4j
@RestController
@RequestMapping("/api/v1/clients")
@Tag(name = "客户端管理", description = "OAuth2客户端注册和管理接口")
@RequiredArgsConstructor
public class ClientRegistrationController {
    
    private final ClientRegistrationService clientRegistrationService;
    
    /**
     * 注册新客户端
     */
    @PostMapping
    @Operation(summary = "注册新客户端")
    @PreAuthorize("hasAuthority('SCOPE_client.write')")
    public ResponseEntity<ClientResponse> registerClient(
            @Valid @RequestBody ClientRegisterRequest request) {
        
        log.info("注册新客户端: clientId={}", request.getClientId());
        ClientResponse response = clientRegistrationService.registerClient(request);
        
        return ResponseEntity
            .created(URI.create("/api/v1/clients/" + response.getClientId()))
            .body(response);
    }
    
    /**
     * 获取客户端列表
     */
    @GetMapping
    @Operation(summary = "获取客户端列表")
    @PreAuthorize("hasAuthority('SCOPE_client.read')")
    public ResponseEntity<Page<ClientResponse>> getClients(
            @ParameterObject Pageable pageable) {
        
        Page<ClientResponse> clients = clientRegistrationService.getClients(pageable);
        return ResponseEntity.ok(clients);
    }
    
    /**
     * 根据ID获取客户端
     */
    @GetMapping("/{clientId}")
    @Operation(summary = "根据ID获取客户端")
    @PreAuthorize("hasAuthority('SCOPE_client.read')")
    public ResponseEntity<ClientResponse> getClient(
            @PathVariable @Parameter(description = "客户端ID") String clientId) {
        
        ClientResponse client = clientRegistrationService.getClient(clientId);
        return ResponseEntity.ok(client);
    }
    
    /**
     * 更新客户端
     */
    @PutMapping("/{clientId}")
    @Operation(summary = "更新客户端")
    @PreAuthorize("hasAuthority('SCOPE_client.write')")
    public ResponseEntity<ClientResponse> updateClient(
            @PathVariable @Parameter(description = "客户端ID") String clientId,
            @Valid @RequestBody ClientRegisterRequest request) {
        
        log.info("更新客户端: clientId={}", clientId);
        ClientResponse response = clientRegistrationService.updateClient(clientId, request);
        return ResponseEntity.ok(response);
    }
    
    /**
     * 删除客户端
     */
    @DeleteMapping("/{clientId}")
    @Operation(summary = "删除客户端")
    @PreAuthorize("hasAuthority('SCOPE_client.delete')")
    public ResponseEntity<Void> deleteClient(
            @PathVariable @Parameter(description = "客户端ID") String clientId) {
        
        log.info("删除客户端: clientId={}", clientId);
        clientRegistrationService.deleteClient(clientId);
        return ResponseEntity.noContent().build();
    }
    
    /**
     * 重置客户端密钥

    @PreAuthorize("hasAuthority('SCOPE_client.write')")

        log.info("批量删除客户端: count={}", clientIds.size());
        clientRegistrationService.batchDeleteClients(clientIds);
        return ResponseEntity.noContent().build();
    }
```

### 3. 创建Service实现

```
package com.aioveu.auth.service.impl;

import com.aioveu.auth.dto.ClientRegisterRequest;
import com.aioveu.auth.dto.ClientResponse;
import com.aioveu.auth.service.ClientRegistrationService;
import com.aioveu.common.exception.BusinessException;
import com.aioveu.common.result.ResultCode;
import lombok.RequiredArgsConstructor;
import lombok.extern.slf4j.Slf4j;
import org.springframework.data.domain.Page;
import org.springframework.data.domain.Pageable;
import org.springframework.security.crypto.password.PasswordEncoder;
import org.springframework.security.oauth2.core.AuthorizationGrantType;
import org.springframework.security.oauth2.core.ClientAuthenticationMethod;
import org.springframework.security.oauth2.server.authorization.client.JdbcRegisteredClientRepository;
import org.springframework.security.oauth2.server.authorization.client.RegisteredClient;
import org.springframework.security.oauth2.server.authorization.settings.ClientSettings;
import org.springframework.security.oauth2.server.authorization.settings.TokenSettings;
import org.springframework.stereotype.Service;
import org.springframework.transaction.annotation.Transactional;
import org.springframework.util.StringUtils;

import java.time.Duration;
import java.time.LocalDateTime;
import java.time.format.DateTimeFormatter;
import java.util.*;
import java.util.stream.Collectors;

@Slf4j
@Service
@RequiredArgsConstructor
public class ClientRegistrationServiceImpl implements ClientRegistrationService {
    
    private final JdbcRegisteredClientRepository registeredClientRepository;
    private final PasswordEncoder passwordEncoder;
    
    @Override
    @Transactional
    public ClientResponse registerClient(ClientRegisterRequest request) {
        // 检查客户端是否已存在
        RegisteredClient existingClient = registeredClientRepository.findByClientId(request.getClientId());
        if (existingClient != null) {
            throw new BusinessException(ResultCode.CLIENT_ALREADY_EXISTS, request.getClientId());
        }
        
        // 生成客户端密钥
        String clientSecret = StringUtils.hasText(request.getClientSecret()) 
            ? request.getClientSecret() 
            : generateRandomSecret();
        String encodedSecret = passwordEncoder.encode(clientSecret);
        
        // 构建RegisteredClient
        RegisteredClient registeredClient = buildRegisteredClient(request, encodedSecret);
        
        // 保存到数据库
        registeredClientRepository.save(registeredClient);
        
        log.info("客户端注册成功: clientId={}", request.getClientId());
        
        return convertToResponse(registeredClient, clientSecret);
    }
    
    @Override
    public ClientResponse getClient(String clientId) {
        RegisteredClient client = getRegisteredClient(clientId);
        return convertToResponse(client, null);
    }
    
    @Override
    public Page<ClientResponse> getClients(Pageable pageable) {
        // 由于JdbcRegisteredClientRepository不提供分页查询，需要自定义实现
        // 这里使用自定义的Repository或JdbcTemplate实现
        throw new UnsupportedOperationException("分页查询需要自定义实现");
    }
    
    @Override
    @Transactional
    public ClientResponse updateClient(String clientId, ClientRegisterRequest request) {
        RegisteredClient existingClient = getRegisteredClient(clientId);
        
        // 检查是否修改了clientId
        if (!existingClient.getClientId().equals(request.getClientId())) {
            // 检查新clientId是否已被占用
            RegisteredClient clientWithNewId = registeredClientRepository.findByClientId(request.getClientId());
            if (clientWithNewId != null && !clientWithNewId.getId().equals(existingClient.getId())) {
                throw new BusinessException(ResultCode.CLIENT_ALREADY_EXISTS, request.getClientId());
            }
        }
        
        // 重新构建客户端
        String clientSecret = null;
        if (StringUtils.hasText(request.getClientSecret())) {
            clientSecret = request.getClientSecret();
        }
        String encodedSecret = clientSecret != null 
            ? passwordEncoder.encode(clientSecret) 
            : existingClient.getClientSecret();
        
        RegisteredClient updatedClient = buildRegisteredClient(request, encodedSecret, existingClient.getId());
        
        // 保存更新
        registeredClientRepository.save(updatedClient);
        
        log.info("客户端更新成功: clientId={}", clientId);
        
        return convertToResponse(updatedClient, clientSecret);
    }
    
    @Override
    @Transactional
    public void deleteClient(String clientId) {
        RegisteredClient client = getRegisteredClient(clientId);
        registeredClientRepository.remove(client);
        log.info("客户端删除成功: clientId={}", clientId);
    }
    
    @Override
    @Transactional
    public ClientResponse resetClientSecret(String clientId) {
        RegisteredClient client = getRegisteredClient(clientId);
        
        // 生成新密钥
        String newSecret = generateRandomSecret();
        String encodedSecret = passwordEncoder.encode(newSecret);
        
        // 更新客户端密钥
        RegisteredClient updatedClient = RegisteredClient.from(client)
            .clientSecret(encodedSecret)
            .build();
        
        registeredClientRepository.save(updatedClient);
        
        log.info("客户端密钥重置成功: clientId={}", clientId);
        
        return convertToResponse(updatedClient, newSecret);
    }
    
    @Override
    @Transactional
    public void toggleClientStatus(String clientId, boolean enabled) {
        // 这里需要扩展RegisteredClient来支持启用/禁用状态
        // 实际实现中可能需要添加自定义字段
        log.info("修改客户端状态: clientId={}, enabled={}", clientId, enabled);
        // 实现细节根据业务需求决定
    }
    
    @Override
    @Transactional
    public void batchDeleteClients(List<String> clientIds) {
        clientIds.forEach(this::deleteClient);
    }
    
    // 私有辅助方法
    private RegisteredClient getRegisteredClient(String clientId) {
        RegisteredClient client = registeredClientRepository.findByClientId(clientId);
        if (client == null) {
            throw new BusinessException(ResultCode.CLIENT_NOT_FOUND, clientId);
        }
        return client;
    }
    
    private RegisteredClient buildRegisteredClient(ClientRegisterRequest request, String encodedSecret) {
        return buildRegisteredClient(request, encodedSecret, UUID.randomUUID().toString());
    }
    
    private RegisteredClient buildRegisteredClient(ClientRegisterRequest request, String encodedSecret, String id) {
        RegisteredClient.Builder builder = RegisteredClient.withId(id)
            .clientId(request.getClientId())
            .clientSecret(encodedSecret)
            .clientName(request.getClientName());
        
        // 设置认证方法
        request.getAuthenticationMethods().forEach(method -> 
            builder.clientAuthenticationMethod(new ClientAuthenticationMethod(method))
        );
        
        // 设置授权类型
        request.getGrantTypes().forEach(grantType -> 
            builder.authorizationGrantType(new AuthorizationGrantType(grantType))
        );
        
        // 设置重定向URI
        request.getRedirectUris().forEach(builder::redirectUri);
        
        // 设置登出重定向URI
        request.getPostLogoutRedirectUris().forEach(builder::postLogoutRedirectUri);
        
        // 设置Scope
        request.getScopes().forEach(builder::scope);
        
        // 客户端设置
        ClientSettings clientSettings = ClientSettings.builder()
            .requireAuthorizationConsent(request.getClientSettings().getRequireAuthorizationConsent())
            .requireProofKey(request.getClientSettings().getRequireProofKey())
            .build();
        builder.clientSettings(clientSettings);
        
        // 令牌设置
        TokenSettings tokenSettings = TokenSettings.builder()
            .accessTokenTimeToLive(Duration.ofSeconds(request.getTokenSettings().getAccessTokenTimeToLiveSeconds()))
            .refreshTokenTimeToLive(Duration.ofSeconds(request.getTokenSettings().getRefreshTokenTimeToLiveSeconds()))
            .reuseRefreshTokens(true)
            .build();
        builder.tokenSettings(tokenSettings);
        
        return builder.build();
    }
    
    private ClientResponse convertToResponse(RegisteredClient client, String rawSecret) {
        ClientResponse response = new ClientResponse();
        response.setId(client.getId());
        response.setClientId(client.getClientId());
        response.setClientName(client.getClientName());
        response.setClientSecret(rawSecret);
        response.setCreatedAt(LocalDateTime.now().format(DateTimeFormatter.ISO_LOCAL_DATE_TIME));
        response.setEnabled(true);
        return response;
    }
    
    private String generateRandomSecret() {
        return UUID.randomUUID().toString().replace("-", "").substring(0, 16);
    }
}
```
