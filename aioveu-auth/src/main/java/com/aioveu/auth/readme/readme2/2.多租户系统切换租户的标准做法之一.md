@Operation(summary = "切换租户")
@PostMapping("/switch-tenant")
public Result<Map<String, Object>> switchTenant(@RequestParam Long tenantId, HttpServletRequest request) {
    
    try {
        // 1. 权限校验
        if (!SecurityUtils.canSwitchTenant()) {
            return Result.failed("无租户切换权限");
        }

        Authentication authentication = SecurityContextHolder.getContext().getAuthentication();
        if (authentication == null || !(authentication.getPrincipal() instanceof SysUserDetails details)) {
            return Result.failed(ResultCode.ACCESS_TOKEN_INVALID);
        }

        // 2. 校验用户是否能访问该租户
        boolean canAccess = tenantFeignClient.canAccessTenant(details.getUserId(), tenantId);
        if (!canAccess) {
            return Result.failed("无权限访问该租户");
        }
        
        // 3. 获取用户在新租户下的权限信息（可选，如果需要更新权限）
        // UserAuthInfoWithTenantId userAuthInfo = tenantFeignClient.getUserAuthInfoWithTenantId(
        //     details.getUsername(), tenantId);
        
        // 4. 更新用户的当前租户（存Redis或数据库，如果后端需要知道）
        updateCurrentTenant(details.getUserId(), tenantId);
        
        // 5. 获取当前Token
        String currentToken = extractTokenFromRequest(request);
        if (currentToken == null) {
            return Result.failed("无法获取当前Token");
        }
        
        // 6. 返回结果
        Map<String, Object> result = new HashMap<>();
        result.put("access_token", currentToken);      // 原Token不变
        result.put("token_type", "Bearer");
        result.put("tenant_id", tenantId);             // 新租户ID
        result.put("user_id", details.getUserId());
        result.put("username", details.getUsername());
        result.put("success", true);
        result.put("timestamp", System.currentTimeMillis());
        
        log.info("用户 {} 切换到租户 {} 成功", details.getUsername(), tenantId);
        return Result.success(result);
        
    } catch (Exception e) {
        log.error("切换租户失败", e);
        return Result.failed("切换失败: " + e.getMessage());
    }
}

// 从请求中提取Token
private String extractTokenFromRequest(HttpServletRequest request) {
    String authHeader = request.getHeader("Authorization");
    if (authHeader != null && authHeader.startsWith("Bearer ")) {
        return authHeader.substring(7);
    }
    return null;
}

// 更新用户当前租户（存Redis）
private void updateCurrentTenant(Long userId, Long tenantId) {
    try {
        String key = "user:current:tenant:" + userId;
        redisTemplate.opsForValue().set(key, tenantId.toString(), 24, TimeUnit.HOURS);
    } catch (Exception e) {
        log.warn("更新用户当前租户失败，但不影响主要功能", e);
    }
}

        }

    if (result.code === 200) {
        const data = result.data;
        
        // ✅ 关键：保存租户ID，Token不变

        return true;
    } else {
        alert('切换失败: ' + result.msg);
        return false;
    }
}

function getRequestHeaders() {
    const headers = {
        'Authorization': 'Bearer ' + localStorage.getItem('access_token'),
        'Content-Type': 'application/json'
    };
    
    // ✅ 关键：添加租户ID到Header
    const currentTenant = localStorage.getItem('current_tenant');
    if (currentTenant) {
        headers['X-Tenant-ID'] = currentTenant;
    }
    
    return headers;
}

public class TenantInterceptor implements HandlerInterceptor {
    
    @Override
    public boolean preHandle(HttpServletRequest request, HttpServletResponse response, Object handler) {
        
        // 获取租户ID
        String tenantIdHeader = request.getHeader("X-Tenant-ID");
        if (tenantIdHeader == null) {
            // 如果没有租户ID，尝试从Token中获取
            tenantIdHeader = extractTenantIdFromToken(request);
        }
        
        if (tenantIdHeader == null) {
            throw new BusinessException("请指定租户");
        }
        
        Long tenantId = Long.parseLong(tenantIdHeader);
        
        // 验证用户是否有权访问该租户

        return true;
