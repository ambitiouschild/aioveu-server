### 方案1：在请求拦截器中统一添加租户ID（推荐）

```
// utils/request.js
import axios from 'axios';

// 创建axios实例
const service = axios.create({
  baseURL: process.env.VUE_APP_BASE_API,
  timeout: 10000
});

// 请求拦截器
service.interceptors.request.use(
  config => {
    // 从本地存储获取租户ID
    const tenantId = localStorage.getItem('current_tenant') || 
                     sessionStorage.getItem('current_tenant');
    
    // 如果有租户ID，添加到Header
    if (tenantId) {
      config.headers['X-Tenant-ID'] = tenantId;
    }
    
    // 如果有Token，添加认证Header
    const token = localStorage.getItem('access_token');
    if (token) {
      config.headers['Authorization'] = 'Bearer ' + token;
    }
    
    return config;
  },
  error => {
    return Promise.reject(error);
  }
);

export default service;
```

// src/composables/useRequest.js
import { ref } from 'vue';
import axios from 'axios';

export function useRequest() {
  const loading = ref(false);
  
  // 获取请求配置
  const getRequestConfig = () => {
    const headers = {
      'Content-Type': 'application/json'
    };
    
    // 添加租户ID
    const tenantId = localStorage.getItem('current_tenant');
    if (tenantId) {
      headers['X-Tenant-ID'] = tenantId;
    }
    
    // 添加Token
    const token = localStorage.getItem('access_token');
    if (token) {
      headers['Authorization'] = 'Bearer ' + token;
    }
    
    return { headers };
  };
  
  // GET请求
  const get = async (url, params = {}) => {
    loading.value = true;
    try {
      const response = await axios.get(url, {
        params,
        ...getRequestConfig()
      });
      return response.data;
    } finally {
      loading.value = false;
    }
  };
  
  // POST请求
  const post = async (url, data = {}) => {
    loading.value = true;
    try {
      const response = await axios.post(url, data, getRequestConfig());
      return response.data;
    } finally {
      loading.value = false;
    }
  };
  
  return {
    loading,
    get,
    post,
    put: async (url, data) => {
      loading.value = true;
      try {
        const response = await axios.put(url, data, getRequestConfig());
        return response.data;
      } finally {
        loading.value = false;
      }
    },
    del: async (url) => {
      loading.value = true;
      try {
        const response = await axios.delete(url, getRequestConfig());
        return response.data;
      } finally {
        loading.value = false;
      }
    }
  };
}

### 方案3：React Hooks 示例

// hooks/useRequest.js
import { useState, useCallback } from 'react';
import axios from 'axios';

export function useRequest() {
  const [loading, setLoading] = useState(false);
  
  // 获取请求头
  const getHeaders = useCallback(() => {
    const headers = {
      'Content-Type': 'application/json'
    };
    
    // 租户ID
    const tenantId = localStorage.getItem('current_tenant');
    if (tenantId) {
      headers['X-Tenant-ID'] = tenantId;
    }
    
    // Token
    const token = localStorage.getItem('access_token');
    if (token) {
      headers['Authorization'] = `Bearer ${token}`;
    }
    
    return headers;
  }, []);
  
  // 统一请求方法
  const request = useCallback(async (method, url, data = null) => {
    setLoading(true);
    try {
      const config = {
        url,
        method,
        headers: getHeaders(),
        baseURL: process.env.REACT_APP_API_URL
      };
      
      if (data) {
        if (method === 'get') {
          config.params = data;
        } else {
          config.data = data;
        }
      }
      
      const response = await axios(config);
      return response.data;
    } finally {
      setLoading(false);
    }
  }, [getHeaders]);
  
  return {
    loading,
    get: (url, params) => request('get', url, params),
    post: (url, data) => request('post', url, data),
    put: (url, data) => request('put', url, data),
    del: (url) => request('delete', url)
  };
}
```

### 1. 租户切换组件

<template>
  <div class="tenant-switcher">
    <el-dropdown @command="handleTenantSwitch">
      <span class="tenant-current">
        {{ currentTenantName }} <i class="el-icon-arrow-down"></i>
      </span>
      <el-dropdown-menu slot="dropdown">
        <el-dropdown-item 
          v-for="tenant in tenantList" 
          :key="tenant.id"
          :command="tenant.id"
          :disabled="tenant.id === currentTenantId"
        >
          {{ tenant.name }}
        </el-dropdown-item>
      </el-dropdown-menu>
    </el-dropdown>
  </div>
</template>

<script>
import { mapState, mapActions } from 'vuex';

export default {
  data() {
    return {
      loading: false
    };
  },
  
  computed: {
    ...mapState('tenant', ['currentTenantId', 'tenantList']),
    
    currentTenantName() {
      const tenant = this.tenantList.find(t => t.id === this.currentTenantId);
      return tenant ? tenant.name : '选择租户';
    }
  },
  
  methods: {
    ...mapActions('tenant', ['switchTenant']),
    
    async handleTenantSwitch(tenantId) {
      if (this.loading) return;
      
      this.loading = true;
      try {
        await this.switchTenant(tenantId);
        
        // 提示成功
        this.$message.success('租户切换成功');
        
        // 刷新页面数据
        this.$store.dispatch('user/getUserInfo');
        this.$store.dispatch('menu/getMenuList');
        
        // 如果有需要，重新加载页面
        // window.location.reload();
        
      } catch (error) {
        this.$message.error(error.message || '切换失败');
      } finally {
        this.loading = false;
      }
    }
  }
};
</script>

      const response = await getTenantListApi();

      return response.data;
    } catch (error) {
      throw error;
    }
  },
  
  // 切换租户

    try {
      const response = await switchTenantApi(tenantId);
      
      // 更新当前租户
      commit('SET_CURRENT_TENANT', tenantId.toString());
      
      // 更新Token（如果有新Token）
      if (response.data.access_token) {
        localStorage.setItem('access_token', response.data.access_token);
        commit('user/SET_TOKEN', response.data.access_token, { root: true });
      }
      
      return response;
    } catch (error) {
      throw error;
    }
  }
};

export default {

  mutations,
  actions

## 几种携带租户ID的方式比较：

| 方式                       | 优点                    | 缺点                | 推荐度 |
| -------------------------- | ----------------------- | ------------------- | ------ |
| **Header (X-Tenant-ID)**   | 标准、灵活、不暴露在URL | 需要前端统一处理    | ★★★★★  |
| **URL Query (?tenantId=)** | 简单直观                | 暴露在URL，不够安全 | ★★★☆☆  |
| **JWT Token 声明**         | 无需每次传递            | Token需要重新生成   | ★★☆☆☆  |
| **Cookie**                 | 自动携带                | 跨域问题，安全性    | ★★☆☆☆  |

## 拦截器自动处理租户ID

```
@Component
public class TenantInterceptor implements HandlerInterceptor {
    
    @Override
    public boolean preHandle(HttpServletRequest request, HttpServletResponse response, 
                           Object handler) throws Exception {
        
        // 从Header获取租户ID
        String tenantIdStr = request.getHeader("X-Tenant-ID");
        
        if (StringUtils.hasText(tenantIdStr)) {
            try {
                Long tenantId = Long.parseLong(tenantIdStr);
                
                // 设置到ThreadLocal
                TenantContextHolder.setTenantId(tenantId);
                
                // 或者更新认证信息
                updateAuthenticationTenantId(tenantId);
                
            } catch (NumberFormatException e) {
                throw new BusinessException("租户ID格式错误");
            }
        } else {
            // 如果没有传租户ID，使用当前用户的默认租户
            Long userId = SecurityUtils.getUserId();
            if (userId != null) {
                String cachedTenant = redisTemplate.opsForValue()
                    .get("user:tenant:" + userId);
                if (cachedTenant != null) {
                    TenantContextHolder.setTenantId(Long.parseLong(cachedTenant));
                }
            }
        }
        
        return true;
    }
    
    @Override
    public void afterCompletion(HttpServletRequest request, HttpServletResponse response, 
                              Object handler, Exception ex) {
        // 清理ThreadLocal
        TenantContextHolder.clear();
    }
}

## 总结要点：

1. **前端必须在每个请求的Header中添加 `X-Tenant-ID`**
2. **使用axios拦截器统一处理**，避免重复代码
3. **租户ID存储在localStorage**，刷新页面后依然存在
4. **切换租户时更新存储的租户ID**
5. **后端通过拦截器统一处理租户ID**

**您的理解完全正确**：前端每次请求都要带租户ID，这是多租户系统正常工作的基础。只要按照上述方案实现，就能确保系统正常工作。
