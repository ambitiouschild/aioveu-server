@Service
@RequiredArgsConstructor
@Slf4j
public class TenantSwitchServiceImpl implements TenantSwitchService {
    
    private final TenantFeignClient tenantFeignClient;
    private final PasswordAuthenticationProvider passwordAuthenticationProvider;
    private final PasswordAuthenticationConverter passwordAuthenticationConverter;
    private final OAuth2AuthorizationService authorizationService;
    private final ServletRequestAttributes servletRequestAttributes;
    
    @Override
    public Result<Authentication> switchTenant(Long tenantId) {
        try {
            // 1. 权限校验
            if (!tenantFeignClient.hasTenantSwitchPermission()) {
                return Result.failed("无权限切换租户");
            }
            
            // 2. 获取当前认证信息
            Authentication currentAuth = SecurityContextHolder.getContext().getAuthentication();
            if (currentAuth == null || !(currentAuth.getPrincipal() instanceof SysUserDetails details)) {
                return Result.failed(ResultCode.ACCESS_TOKEN_INVALID);
            }
            
            // 3. 校验用户是否能访问该租户
            boolean canAccess = tenantFeignClient.canAccessTenant(details.getUserId(), tenantId);
            if (!canAccess) {
                return Result.failed("无权限访问该租户");
            }
            
            // 4. 获取用户在新租户下的权限信息
            UserAuthInfoWithTenantId userAuthInfo = tenantFeignClient.getUserAuthInfoWithTenantId(
                details.getUsername(), tenantId);
            
            // 5. 构建新的用户详情
            SysUserDetails newDetails = buildNewUserDetails(details, userAuthInfo, tenantId);
            
            // 6. 模拟登录请求
            HttpServletRequest request = buildLoginRequest(newDetails);
            
            // 7. 通过 Converter 创建认证令牌
            Authentication passwordAuth = passwordAuthenticationConverter.convert(request);
            if (passwordAuth == null) {
                throw new RuntimeException("构建认证请求失败");
            }
            
            // 8. 调用认证提供者（复用登录流程）
            Authentication tokenResult = passwordAuthenticationProvider.authenticate(passwordAuth);
            
            log.info("用户 {} 切换到租户 {} 成功", details.getUsername(), tenantId);
            return Result.success(tokenResult);
            
        } catch (Exception e) {
            log.error("切换租户失败", e);
            return Result.failed("切换失败: " + e.getMessage());
        }
    }
    
    /**
     * 构建新的用户详情
     */
    private SysUserDetails buildNewUserDetails(
            SysUserDetails oldDetails, 
            UserAuthInfoWithTenantId userAuthInfo,
            Long tenantId) {
        
        SysUserDetails newDetails = new SysUserDetails(userAuthInfo);
        newDetails.setUserId(oldDetails.getUserId());
        newDetails.setUsername(oldDetails.getUsername());
        newDetails.setDeptId(oldDetails.getDeptId());
        newDetails.setDataScopes(oldDetails.getDataScopes());
        newDetails.setTenantId(tenantId);
        newDetails.setCanSwitchTenant(oldDetails.getCanSwitchTenant());
        
        return newDetails;
    }
    
    /**
     * 构建模拟的登录 HTTP 请求
     */
    private HttpServletRequest buildLoginRequest(SysUserDetails userDetails) {
        // 获取当前请求上下文
        ServletRequestAttributes attributes = (ServletRequestAttributes) 
            RequestContextHolder.getRequestAttributes();
        
        if (attributes == null) {
            throw new RuntimeException("无法获取请求上下文");
        }
        
        HttpServletRequest originalRequest = attributes.getRequest();
        
        // 创建包装请求，修改参数
        return new HttpServletRequestWrapper(originalRequest) {
            private final Map<String, String[]> modifiedParams = new HashMap<>();
            
            {
                // 复制原参数
                Map<String, String[]> originalParams = originalRequest.getParameterMap();
                modifiedParams.putAll(originalParams);
                
                // 修改登录参数
                modifiedParams.put("grant_type", new String[]{"password"});
                modifiedParams.put("username", new String[]{userDetails.getUsername()});
                modifiedParams.put("password", new String[]{"[TENANT_SWITCH]"}); // 特殊标记
                modifiedParams.put("tenant_id", new String[]{String.valueOf(userDetails.getTenantId())});
                
                // 添加客户端认证（从配置读取）
                modifiedParams.put("client_id", new String[]{"your-client-id"});
                modifiedParams.put("client_secret", new String[]{"your-client-secret"});
            }
            
            @Override
            public String getParameter(String name) {
                String[] values = modifiedParams.get(name);
                return values != null && values.length > 0 ? values[0] : null;
            }
            
            @Override
            public Map<String, String[]> getParameterMap() {
                return modifiedParams;
            }
            
            @Override
            public String[] getParameterValues(String name) {
                return modifiedParams.get(name);
            }
        };
    }
}

@Component

public class SimpleTenantSwitchService {
    
    private final RestTemplate restTemplate;
    private final TenantFeignClient tenantFeignClient;
    
    public Result<AuthenticationToken> switchTenantSimple(Long tenantId) {
        try {
            // 1. 获取当前用户
            Authentication auth = SecurityContextHolder.getContext().getAuthentication();
            SysUserDetails details = (SysUserDetails) auth.getPrincipal();
            
            // 2. 构建登录请求参数
            MultiValueMap<String, String> params = new LinkedMultiValueMap<>();
            params.add("grant_type", "password");
            params.add("username", details.getUsername());
            params.add("password", "[TENANT_SWITCH]");
            params.add("scope", "read write");
            params.add("tenant_id", String.valueOf(tenantId));
            
            // 3. 设置客户端认证
            HttpHeaders headers = new HttpHeaders();
            headers.setBasicAuth("your-client-id", "your-client-secret");
            headers.setContentType(MediaType.APPLICATION_FORM_URLENCODED);
            
            // 4. 调用 OAuth2 令牌端点
            HttpEntity<MultiValueMap<String, String>> request = new HttpEntity<>(params, headers);
            ResponseEntity<Map> response = restTemplate.postForEntity(
                "http://your-auth-server/oauth2/token", 
                request, 
                Map.class
            );
            
            // 5. 解析响应
            Map<String, Object> tokenMap = response.getBody();
            AuthenticationToken token = AuthenticationToken.builder()
                .accessToken((String) tokenMap.get("access_token"))
                .refreshToken((String) tokenMap.get("refresh_token"))
                .tokenType("Bearer")
                .expiresIn((Integer) tokenMap.get("expires_in"))
                .build();
            
            return Result.success(token);
            
        } catch (Exception e) {
            return Result.failed("切换失败: " + e.getMessage());
        }
    }
}

import javax.servlet.http.HttpServletRequest;
import javax.servlet.http.HttpServletRequestWrapper;
import java.util.*;

/**
 * 参数可修改的 HttpServletRequest 包装类
 */
public class ModifiableHttpServletRequest extends HttpServletRequestWrapper {
    
    private final Map<String, String[]> modifiedParameters = new HashMap<>();
    
    public ModifiableHttpServletRequest(HttpServletRequest request) {
        super(request);
        modifiedParameters.putAll(request.getParameterMap());
    }
    
    public void setParameter(String name, String value) {
        modifiedParameters.put(name, new String[]{value});
    }
    
    public void setParameter(String name, String[] values) {
        modifiedParameters.put(name, values);
    }
    
    @Override
    public String getParameter(String name) {
        String[] values = modifiedParameters.get(name);
        return values != null && values.length > 0 ? values[0] : null;
    }
    
    @Override
    public Map<String, String[]> getParameterMap() {
        return Collections.unmodifiableMap(modifiedParameters);
    }
    
    @Override
    public Enumeration<String> getParameterNames() {
        return Collections.enumeration(modifiedParameters.keySet());
    }
    
    @Override
    public String[] getParameterValues(String name) {
        return modifiedParameters.get(name);
    }
}

## ✅ 总结

1. **修改认证提供者**：添加对 `[TENANT_SWITCH]`密码的特殊处理
2. **构建模拟请求**：创建包含用户信息和租户ID的HTTP请求
3. **调用认证流程**：通过 `passwordAuthenticationConverter.convert()`和 `passwordAuthenticationProvider.authenticate()`生成新令牌
4. **返回结果**：将生成的令牌返回给前端

这样生成的 Token 与登录完全一致，因为走的是完全相同的流程！
