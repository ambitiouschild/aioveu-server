```
@Service
@RequiredArgsConstructor
public class TenantSwitchService {
    
    private final AuthenticationManager authenticationManager;
    private final PasswordAuthenticationProvider passwordAuthenticationProvider; // 直接使用登录的 Provider
    
    /**
     * 切换租户（内部模拟登录）
     */
    public AuthenticationToken switchTenant(String originalToken, String newTenantId) {
        // 1. 从原 Token 中提取用户信息
        UserInfo userInfo = extractUserInfo(originalToken);
        
        // 2. 构建认证请求（模拟登录请求）
        PasswordAuthenticationToken authentication = buildAuthenticationRequest(userInfo, newTenantId);
        
        // 3. 直接调用登录的 AuthenticationProvider
        OAuth2AccessTokenAuthenticationToken result = 
            (OAuth2AccessTokenAuthenticationToken) passwordAuthenticationProvider.authenticate(authentication);
        
        // 4. 构建返回结果
        return convertToTokenResponse(result);
    }
    
    /**
     * 从原 Token 提取用户信息
     */
    private UserInfo extractUserInfo(String token) {
        // 移除 Bearer 前缀
        if (token.startsWith("Bearer ")) {
            token = token.substring(7);
        }
        
        // 解析 JWT
        JWT jwt = JWTUtil.parseToken(token);
        JSONObject payload = jwt.getPayloads();
        
        UserInfo userInfo = new UserInfo();
        userInfo.setUsername(payload.getStr("sub"));
        userInfo.setUserId(payload.getLong(JwtClaimConstants.USER_ID));
        userInfo.setDeptId(payload.getLong(JwtClaimConstants.DEPT_ID));
        userInfo.setOriginalTenantId(payload.getStr(JwtClaimConstants.TENANT_ID));
        
        // 获取权限
        JSONArray authorities = payload.getJSONArray(JwtClaimConstants.AUTHORITIES);
        if (authorities != null) {
            userInfo.setAuthorities(new HashSet<>(authorities.toList(String.class)));
        }
        
        return userInfo;
    }
    
    /**
     * 构建认证请求（与前端登录完全一致）
     */
    private PasswordAuthenticationToken buildAuthenticationRequest(UserInfo userInfo, String newTenantId) {
        Map<String, Object> additionalParameters = new HashMap<>();
        additionalParameters.put("username", userInfo.getUsername());
        additionalParameters.put("password", "[PROTECTED]"); // 密码需要特殊处理
        
        // 这里需要根据实际情况获取密码
        // 方案1：从安全上下文获取（如果用户已登录）
        // 方案2：查询数据库（需要密码加密匹配）
        // 方案3：使用特殊标记绕过密码验证
        
        // 添加租户信息
        additionalParameters.put("tenant_id", newTenantId);
        
        // 构建与登录完全相同的请求
        return new PasswordAuthenticationToken(
            new UsernamePasswordAuthenticationToken(userInfo.getUsername(), ""),
            additionalParameters,
            userInfo.getAuthorities()
        );
    }
    
    /**
     * 转换认证结果
     */
    private AuthenticationToken convertToTokenResponse(OAuth2AccessTokenAuthenticationToken authResult) {
        OAuth2AccessToken accessToken = authResult.getAccessToken();
        OAuth2RefreshToken refreshToken = authResult.getRefreshToken();
        
        return AuthenticationToken.builder()
            .accessToken(accessToken.getTokenValue())
            .refreshToken(refreshToken != null ? refreshToken.getTokenValue() : null)
            .tokenType("Bearer")
            .expiresIn((int) ((accessToken.getExpiresAt().getTime() - System.currentTimeMillis()) / 1000))
            .build();
    }
}
```

### 2. **更简单的方案：直接调用登录接口**

```
@Service
public class SimpleTenantSwitchService {
    
    private final RestTemplate restTemplate = new RestTemplate();
    
    /**
     * 直接内部调用登录接口（最简单！）
     */
    public AuthenticationToken switchTenantSimple(String originalToken, String newTenantId) {
        // 1. 解析原 Token 获取用户名
        String username = extractUsername(originalToken);
        
        // 2. 获取用户密码（需要根据你的安全策略）
        String password = getPasswordFromSecurityContext(username);
        
        // 3. 构建登录请求体（与前端登录完全一致）
        Map<String, String> requestBody = new HashMap<>();
        requestBody.put("grant_type", "password");
        requestBody.put("username", username);
        requestBody.put("password", password);
        requestBody.put("client_id", "your-client-id");
        requestBody.put("client_secret", "your-client-secret");
        requestBody.put("tenant_id", newTenantId); // 添加租户参数
        
        // 4. 设置请求头
        HttpHeaders headers = new HttpHeaders();
        headers.setContentType(MediaType.APPLICATION_FORM_URLENCODED);
        headers.setBasicAuth("your-client-id", "your-client-secret");
        
        // 5. 调用登录接口
        HttpEntity<MultiValueMap<String, String>> request = createRequest(requestBody, headers);
        
        ResponseEntity<Map> response = restTemplate.postForEntity(
            "http://localhost:8080/oauth2/token", // 你的登录接口
            request,
            Map.class
        );
        
        // 6. 解析响应
        Map<String, Object> tokenMap = response.getBody();
        return AuthenticationToken.builder()
            .accessToken((String) tokenMap.get("access_token"))
            .refreshToken((String) tokenMap.get("refresh_token"))
            .tokenType("Bearer")
            .expiresIn((Integer) tokenMap.get("expires_in"))
            .build();
    }
    
    /**
     * 辅助方法：将 Map 转换为 MultiValueMap
     */
    private HttpEntity<MultiValueMap<String, String>> createRequest(
            Map<String, String> params, HttpHeaders headers) {
        
        MultiValueMap<String, String> multiValueMap = new LinkedMultiValueMap<>();
        params.forEach(multiValueMap::add);
        
        return new HttpEntity<>(multiValueMap, headers);
    }
    
    /**
     * 获取用户密码（需要根据你的实现调整）
     */
    private String getPasswordFromSecurityContext(String username) {
        // 方案1：从 ThreadLocal 或 SecurityContext 获取
        Authentication authentication = SecurityContextHolder.getContext().getAuthentication();
        if (authentication != null && authentication.getCredentials() != null) {
            return authentication.getCredentials().toString();
        }
        
        // 方案2：从缓存获取（如果登录时缓存了）
        String cacheKey = "user:pwd:" + username;
        return (String) redisTemplate.opsForValue().get(cacheKey);
        
        // 方案3：特殊标记（需要修改认证逻辑支持）
        // return "[TENANT_SWITCH]"; // 在认证提供者中特殊处理
    }
}
```

### 3. **修改认证提供者支持免密码切换**

```
// 在 PasswordAuthenticationProvider 中添加特殊逻辑
public class PasswordAuthenticationProvider implements AuthenticationProvider {
    
    @Override
    public Authentication authenticate(Authentication authentication) throws AuthenticationException {
        PasswordAuthenticationToken passwordAuth = (PasswordAuthenticationToken) authentication;
        Map<String, Object> additionalParams = passwordAuth.getAdditionalParameters();
        
        String password = (String) additionalParams.get("password");
        String tenantId = (String) additionalParams.get("tenant_id");
        
        // 特殊处理：如果是租户切换请求，跳过密码验证
        if ("[TENANT_SWITCH]".equals(password)) {
            // 从 Token 或其他地方获取用户信息
            String username = (String) additionalParams.get("username");
            UserDetails userDetails = loadUserForTenantSwitch(username, tenantId);
            
            UsernamePasswordAuthenticationToken usernamePasswordAuth = 
                new UsernamePasswordAuthenticationToken(userDetails, null, userDetails.getAuthorities());
            
            // 继续走后续的令牌生成流程
            return processAuthentication(usernamePasswordAuth, passwordAuth);
        }
        
        // 正常登录流程
        // ... 你的原有代码
    }
    
    private UserDetails loadUserForTenantSwitch(String username, String tenantId) {
        // 加载用户信息，并设置新的租户
        SysUserDetails userDetails = userDetailsService.loadUserByUsername(username);
        userDetails.setTenantId(tenantId);
        
        // 重新加载该租户下的权限
        userDetails.setAuthorities(loadPermissionsForTenant(username, tenantId));
        
        return userDetails;
    }
}
```
