seata配置模板



```
# Seata 配置
seata: 
  # 开启 Seata
  enabled: true
  #第三步：pms 注册到 Seata Server ✅  pms的Seata客户端配置
  # 这个ID是pms在Seata中的标识
  application-id: pms-service
  # 事务分组名称
  # 这是pms所属的事务组，多个微服务可以用同一个分组
  tx-service-group: default_tx_group
  service:
    # 事务分组映射，根据分组名称匹配到集群
    vgroup-mapping:
      # 事务分组名称：集群名称
      # 告诉pms：default_tx_group分组对应的Seata集群是"default"
      default_tx_group: default

  #第一步：Seata Server 注册到 Nacos ✅
  #容器创建时已经注册

  # pms通过application.yaml注册进nacos
  #registry 配置是读取注册配置，不是进行注册**
  #第二步：pms 通过seata的注册信息从 Nacos 发现 Seata Server ✅ ，这里告诉pms要查找的seata服务名
  registry:
    type: nacos
    nacos:
      # ❌ 这里理解有误！
      # 这个不是pms注册到Nacos，而是告诉pms：
      # "去Nacos里找名为'aioveu-seata-server'的服务"（即Seata Server）
      application: ${nacos.application}
      server-addr: ${nacos.server-addr}
      namespace: ${nacos.namespace} 
      group: ${nacos.group}
      username: ${nacos.username}
      password: ${nacos.password}

  #config 配置是读取中心化配置，不是设置本地配置**
    # 5. 如何获取Seata配置（从Nacos读取）  告诉 pms 去哪里读配置
  config:
    type: nacos
    nacos:
      server-addr: ${nacos.server-addr}
      namespace: ${nacos.namespace}
      group: ${nacos.group}
      data-id: ${nacos.data-id}
      username: ${nacos.username}
      password: ${nacos.password}
```
