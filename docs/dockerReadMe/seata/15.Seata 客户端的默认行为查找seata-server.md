public class NacosRegistryServiceImpl implements RegistryService {
    
    // 默认的服务名
    private static final String DEFAULT_SERVICE_NAME = "seata-server";
    
    public String getServiceName(String key) {
        // 这里会解析key，如果key中不包含服务名，就用默认的
        if (key.contains(":")) {
            return key.split(":")[0];
        }
        return DEFAULT_SERVICE_NAME;  // ← 返回默认值！
    }
}
