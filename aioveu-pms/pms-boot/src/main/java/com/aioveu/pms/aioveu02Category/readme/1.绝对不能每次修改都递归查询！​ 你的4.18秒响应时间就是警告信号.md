## ✅ 正确的解决方案

### 方案1：**缓存分类树**（推荐）

```
@Service
public class CategoryService {
    
    @Autowired
    private RedisTemplate<String, Object> redisTemplate;
    
    private static final String CATEGORY_TREE_KEY = "category:tree";
    
    // 获取分类树（带缓存）
    public CategoryTree getCategoryTree() {
        // 1. 先查缓存
        CategoryTree tree = (CategoryTree) redisTemplate.opsForValue().get(CATEGORY_TREE_KEY);
        if (tree != null) {
            return tree;
        }
        
        // 2. 缓存没有，从数据库查
        tree = buildCategoryTreeFromDB();
        
        // 3. 放入缓存
        redisTemplate.opsForValue().set(CATEGORY_TREE_KEY, tree, 1, TimeUnit.HOURS);
        
        return tree;
    }
    
    // 修改分类时，清除缓存
    @Transactional
    public void updateCategory(Category category) {
        // 1. 更新数据库
        categoryMapper.updateById(category);
        
        // 2. 删除缓存
        redisTemplate.delete(CATEGORY_TREE_KEY);
    }
}
```

### 方案2：**使用闭包表（Closure Table）**

```
-- 闭包表设计
CREATE TABLE category_closure (
    ancestor_id BIGINT NOT NULL,   -- 祖先ID
    descendant_id BIGINT NOT NULL, -- 后代ID
    depth INT NOT NULL,            -- 深度
    PRIMARY KEY (ancestor_id, descendant_id)
);

-- 查询整个树（一次查询！）
SELECT c.* 
FROM category c
JOIN category_closure cc ON c.id = cc.descendant_id
WHERE cc.ancestor_id = 1  -- 根节点ID
ORDER BY cc.depth;
```

### 方案3：**物化路径（Materialized Path）**

```
-- 物化路径设计
CREATE TABLE category (
    id BIGINT PRIMARY KEY,
    name VARCHAR(100),
    path VARCHAR(1000),  -- 如: 1/3/5/7
    depth INT
);

-- 查询子树（一次查询！）
SELECT * FROM category 
WHERE path LIKE '1/3/%'  -- 查找节点3的所有子孙
ORDER BY path;
```
