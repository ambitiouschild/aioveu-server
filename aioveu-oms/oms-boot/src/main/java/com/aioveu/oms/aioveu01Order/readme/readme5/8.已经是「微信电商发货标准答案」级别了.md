@Override
public JsonNode uploadShipping(String orderSn, ShipOrderDTO dto) {

    OmsOrder order = getByOrderNo(orderSn);
    if (order == null) {
        throw new BusinessException("订单不存在");
    }

    Long orderId = order.getId();

    OmsOrderDelivery delivery =
            omsOrderDeliveryService.selectByOrderId(orderId);

    if (delivery == null) {
        throw new BusinessException("发货信息不存在");
    }

    // ✅ 幂等：已同步微信直接跳过
    if (delivery.getDeliveryStatus() == 0) {
        log.warn("微信发货已同步，跳过 orderSn={}", orderSn);
        return buildSkippedResult();
    }

    // ✅ 回填物流信息（人工）
    delivery.setDeliveryCompany(dto.getLogisticsCompany());
    delivery.setDeliverySn(dto.getTrackingNo());
    delivery.setRemark(dto.getRemark());
    delivery.setDeliveryStatus(0);
    delivery.setDeliveryTime(new Date());

    omsOrderDeliveryService.updateById(delivery);

    // ✅ 微信发货
    return doUploadShipping(order, delivery);
}

@Override
public JsonNode uploadShipping(String orderSn) {

    OmsOrder order = getByOrderNo(orderSn);
    if (order == null) {
        throw new BusinessException("订单不存在");
    }

    OmsOrderDelivery delivery =
            omsOrderDeliveryService.selectByOrderId(order.getId());

    if (delivery == null) {
        throw new BusinessException("发货信息不存在");
    }

    if (StrUtil.hasBlank(
            delivery.getDeliveryCompany(),
            delivery.getDeliverySn())) {
        throw new BusinessException("物流信息不完整");
    }

    if (delivery.getDeliveryStatus() == 0) {
        log.info("微信发货已同步，跳过 orderSn={}", orderSn);
        return buildSkippedResult();
    }

    return doUploadShipping(order, delivery);
}

# 五、✅ 微信发货「真正干活的方法（核心）」

```
private JsonNode doUploadShipping(OmsOrder order, OmsOrderDelivery delivery) {

    // 商品描述
    List<OmsOrderItem> items =
            omsOrderItemService.listByOrderId(order.getId());

    String itemDesc = items.stream()
            .map(i -> i.getSpuName()
                    + Optional.ofNullable(i.getSkuName())
                    .map(sku -> "【" + sku + "】")
                    .orElse("")
                    + " x" + i.getQuantity())
            .collect(Collectors.joining("；"));

    // openId
    String openId = memberFeignClient
            .getOpenIdByMemberId(order.getMemberId())
            .getData();

    if (StrUtil.isBlank(openId)) {
        throw new BusinessException("用户openid不存在");
    }

    // 组装微信参数
    ObjectNode body = weChatApiClient.createObjectNode();

    ObjectNode orderKey = body.putObject("order_key");
    orderKey.put("order_number_type", 2);
    orderKey.put("transaction_id", order.getTransactionId());

    body.put("logistics_type", 1);
    body.put("delivery_mode", 1);

    ArrayNode shippingList = body.putArray("shipping_list");
    ObjectNode shippingItem = shippingList.addObject();
    shippingItem.put("tracking_no", delivery.getDeliverySn());
    shippingItem.put("express_company", delivery.getDeliveryCompany());
    shippingItem.put("item_desc", itemDesc);

    body.put("upload_time", OffsetDateTime.now().toString());

    ObjectNode payer = body.putObject("payer");
    payer.put("openid", openId);

    // 调用微信
    JsonNode result = weChatApiClient.uploadShippingInfo(
            order.getClientId(),
            body
    );

    // ✅ 成功才标记
    if (result.has("errcode") && result.get("errcode").asInt() == 0) {
        delivery.setDeliveryStatus(0);
        delivery.setDeliveryTime(new Date());
        omsOrderDeliveryService.updateById(delivery);
    }

    return result;
}
```
