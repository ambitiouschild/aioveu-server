### ✅ 2️⃣ 前端正确写法（照抄）

```js
const handleSubmit = async () => {
  uni.showLoading({ title: "提交中..." });

  try {
    const res = await submitOrder(submitData);

    if (res.code !== "00000") {
      throw new Error(res.message || "提交失败");
    }

    const { orderSn, paymentNo, paymentAmount } = res.data;

    console.log("✅ 订单提交成功", {
      orderSn,
      paymentNo,
      paymentAmount,
    });

    uni.redirectTo({
      url:
        `/packageD/pages/money/pay` +
        `?orderSn=${orderSn}` +
        `&paymentNo=${paymentNo}` +
        `&paymentAmount=${paymentAmount}`,
    });
  } catch (err) {
    uni.showToast({
      title: err.message || "提交失败",
      icon: "none",
    });
  } finally {
    uni.hideLoading();
  }
};
```

✅ orderSn 从后端来
✅ paymentNo 从后端来
✅ 前端不再“脑补订单号”
