public Object payOrder(OrderPaymentForm form) {

    // 金额校验（用 PayOrder）

            "支付金额不匹配");

    // 分布式锁
    RLock lock = redissonClient.getLock(ORDER_LOCK_PREFIX + order.getOrderSn());
    lock.lock();
    try {
        return processPayment(order, payOrder, form.getPaymentMethod());
    } finally {
        lock.unlock();
    }
}
