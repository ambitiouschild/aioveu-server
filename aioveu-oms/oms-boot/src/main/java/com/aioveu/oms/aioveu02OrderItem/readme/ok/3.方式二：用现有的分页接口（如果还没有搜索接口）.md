**问得好！** 获取列表有两种方式：

## 方式一：后端提供专门的搜索接口（推荐）

这是标准做法，后端提供一个搜索订单的接口：

### 1. 后端搜索接口

```
@RestController
@RequestMapping("/api/orders")
@RequiredArgsConstructor
public class OrderController {
    
    private final OrderService orderService;
    
    /**
     * 搜索订单（用于选择器）
     */
    @GetMapping("/search")
    public ApiResult<List<OrderOptionVO>> searchOrders(
            @RequestParam(required = false) String keyword,
            @RequestParam(defaultValue = "1") int page,
            @RequestParam(defaultValue = "20") int size) {
        
        List<OrderOptionVO> list = orderService.searchOrders(keyword, page, size);
        return ApiResult.success(list);
    }
}

// OrderOptionVO.java
@Data
public class OrderOptionVO {
    private Long id;           // 订单ID
    private String orderSn;    // 订单号
    private String customerName;  // 客户名
    private BigDecimal totalAmount;  // 订单金额
    private Integer orderStatus;  // 订单状态
    
    // 显示用
    public String getDisplayText() {
        return orderSn + " - " + customerName + " - ¥" + totalAmount;
    }
}
```

### 2. 后端Service实现

```
@Service
@RequiredArgsConstructor
public class OrderServiceImpl implements OrderService {
    
    private final OrderMapper orderMapper;
    
    @Override
    public List<OrderOptionVO> searchOrders(String keyword, int page, int size) {
        PageHelper.startPage(page, size);
        
        // 构建查询条件
        OrderQueryDTO query = new OrderQueryDTO();
        if (StringUtils.isNotBlank(keyword)) {
            query.setKeyword(keyword);
        }
        query.setDeleted(0);  // 只查未删除的
        
        List<Order> orders = orderMapper.selectList(query);
        
        return orders.stream()
            .map(order -> {
                OrderOptionVO vo = new OrderOptionVO();
                vo.setId(order.getId());
                vo.setOrderSn(order.getOrderSn());
                vo.setCustomerName(order.getCustomerName());
                vo.setTotalAmount(order.getTotalAmount());
                vo.setOrderStatus(order.getOrderStatus());
                return vo;
            })
            .collect(Collectors.toList());
    }
}
```

### 3. 前端调用

```
<script setup lang="ts">
import { ref } from 'vue';
import { searchOrders } from '@/api/order';

// 订单选项列表
const orderOptions = ref<any[]>([]);
const orderLoading = ref(false);

// 搜索订单
async function searchOrdersHandler(query: string) {
  if (!query || query.trim() === '') {
    orderOptions.value = [];
    return;
  }
  
  orderLoading.value = true;
  try {
    const res = await searchOrders({
      keyword: query,
      page: 1,
      size: 20
    });
    
    if (res.code === 200) {
      orderOptions.value = res.data;
    } else {
      orderOptions.value = [];
    }
  } catch (error) {
    console.error('搜索订单失败:', error);
    orderOptions.value = [];
  } finally {
    orderLoading.value = false;
  }
}
</script>
```

## 方式二：用现有的分页接口（如果还没有搜索接口）

如果后端暂时没有专门的搜索接口，可以用现有的分页接口：

### 1. 前端调整

```
// 使用已有的订单分页接口
async function searchOrdersHandler(query: string) {
  if (!query || query.trim() === '') {
    orderOptions.value = [];
    return;
  }
  
  orderLoading.value = true;
  try {
    // 调用已有的订单分页接口
    const res = await getOrderPage({
      orderSn: query,  // 按订单号搜索
      pageNum: 1,
      pageSize: 20
    });
    
    if (res.code === 200) {
      orderOptions.value = res.data.list.map((order: any) => ({
        id: order.id,
        orderSn: order.orderSn,
        customerName: order.customerName,
        totalAmount: order.totalAmount
      }));
    }
  } catch (error) {
    console.error('搜索订单失败:', error);
    orderOptions.value = [];
  } finally {
    orderLoading.value = false;
  }
}
```
