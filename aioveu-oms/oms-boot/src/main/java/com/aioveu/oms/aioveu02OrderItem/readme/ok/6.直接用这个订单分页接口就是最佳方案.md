## 前端实现代码

```
<template>
  <el-select
    v-model="selectedOrderId"
    filterable
    remote
    :remote-method="searchOrders"
    :loading="orderLoading"
    placeholder="输入订单号搜索"
    style="width: 100%"
    @change="handleOrderChange"
  >
    <el-option
      v-for="order in orderOptions"
      :key="order.id"
      :label="formatOrderLabel(order)"
      :value="order.id"
    >
      <div style="display: flex; justify-content: space-between; align-items: center;">
        <span>{{ order.orderSn }}</span>
        <div style="display: flex; flex-direction: column; align-items: flex-end;">
          <span style="color: #909399; font-size: 12px">
            ¥{{ (order.totalAmount / 100).toFixed(2) }}
          </span>
          <el-tag 
            v-if="order.status" 
            :type="getStatusType(order.status)"
            size="small"
            style="margin-top: 2px;"
          >
            {{ getStatusText(order.status) }}
          </el-tag>
        </div>
      </div>
    </el-option>
  </el-select>
  
  <!-- 显示已选订单信息 -->
  <div v-if="selectedOrderInfo" style="margin-top: 8px; color: #666; font-size: 12px">
    已选择: {{ selectedOrderInfo.orderSn }} 
    <span v-if="selectedOrderInfo.totalAmount">
      (¥{{ (selectedOrderInfo.totalAmount / 100).toFixed(2) }})
    </span>
  </div>
</template>

<script setup lang="ts">
import { ref, watch } from 'vue';
import { getOrderPage } from '@/api/order'; // 假设这是你的订单接口

const props = defineProps({
  modelValue: Number, // 选中的订单ID
});

const emit = defineEmits(['update:modelValue', 'change']);

const selectedOrderId = ref<number | null>(props.modelValue || null);
const selectedOrderInfo = ref<any>(null);
const orderOptions = ref<any[]>([]);
const orderLoading = ref(false);

// 监听外部传入的值
watch(() => props.modelValue, (newVal) => {
  selectedOrderId.value = newVal || null;
});

// 搜索订单
const searchOrders = async (keyword: string) => {
  if (!keyword || keyword.trim() === '') {
    orderOptions.value = [];
    return;
  }
  
  orderLoading.value = true;
  try {
    const res = await getOrderPage({
      keywords: keyword.trim(),
      pageNum: 1,
      pageSize: 20, // 限制数量
      // 可以添加其他过滤条件
      // status: 1, // 只查已支付订单等
    });
    
    if (res.code === 200 && res.data && res.data.list) {
      orderOptions.value = res.data.list;
      
      // 如果当前选中的订单不在搜索结果中，自动选中第一个
      if (orderOptions.value.length > 0 && !selectedOrderId.value) {
        const firstOrder = orderOptions.value[0];
        selectedOrderId.value = firstOrder.id;
        selectedOrderInfo.value = firstOrder;
        emit('update:modelValue', firstOrder.id);
        emit('change', { id: firstOrder.id, order: firstOrder });
      }
    } else {
      orderOptions.value = [];
    }
  } catch (error) {
    console.error('搜索订单失败:', error);
    orderOptions.value = [];
    ElMessage.error('搜索失败');
  } finally {
    orderLoading.value = false;
  }
};

// 订单选择变化
const handleOrderChange = (orderId: number) => {
  const selectedOrder = orderOptions.value.find(order => order.id === orderId);
  selectedOrderInfo.value = selectedOrder || null;
  
  emit('update:modelValue', orderId);
  emit('change', { 
    id: orderId, 
    order: selectedOrder 
  });
};

// 格式化显示标签
const formatOrderLabel = (order: any) => {
  let label = order.orderSn;
  if (order.totalAmount) {
    label += ` - ¥${(order.totalAmount / 100).toFixed(2)}`;
  }
  return label;
};

// 订单状态显示
const getStatusType = (status: number) => {
  const map: Record<number, string> = {
    0: 'info',    // 待支付
    1: 'success', // 已支付
    2: 'warning', // 已发货
    3: 'success', // 已完成
    4: 'danger',  // 已取消
  };
  return map[status] || 'info';
};

const getStatusText = (status: number) => {
  const map: Record<number, string> = {
    0: '待支付',
    1: '已支付',
    2: '已发货',
    3: '已完成',
    4: '已取消',
  };
  return map[status] || '未知';
};

// 外部可以调用此方法设置值
const setOrder = (order: any) => {
  if (order && order.id) {
    selectedOrderId.value = order.id;
    selectedOrderInfo.value = order;
    // 如果订单不在当前选项列表中，添加到列表
    if (!orderOptions.value.find(o => o.id === order.id)) {
      orderOptions.value.unshift(order);
    }
  }
};

// 暴露方法给父组件
defineExpose({
  setOrder,
  searchOrders,
});
</script>
```
