<select id="getOrderStatusCounts" resultType="com.xxx.OrderStatusCountDTO">

    <where>

## 三、Service（✅ 类型安全、无强转、无 ClassCastException）

### ✅ `getOrderStatistics` 最终正确版

```java
@Override
public OrderStatisticsVO getOrderStatistics(OrderPageQuery queryParams) {

    OrderStatisticsVO result = new OrderStatisticsVO();

    try {
        // 1. 查询状态统计
        List<OrderStatusCountDTO> dtoList =
                this.baseMapper.getOrderStatusCounts(queryParams);

        Map<String, Integer> statusCounts = new HashMap<>();
        int total = 0;

        for (OrderStatusCountDTO dto : dtoList) {
            OrderStatusEnum statusEnum = OrderStatusEnum.of(dto.getStatus());
            statusCounts.put(statusEnum.name(), dto.getCount());
            total += dto.getCount();
        }

        result.setStatusCounts(statusCounts);
        result.setTotal(total);

        // 2. 今日订单数
        result.setTodayOrderCount(getTodayOrderCount(queryParams));

        // 3. 待处理订单数
        result.setPendingCount(getPendingOrderCount(queryParams));

        // 4. 今日收入
        result.setTodayIncome(getTodayIncome(queryParams));

        log.info("订单统计结果: {}", result);
        return result;

    } catch (Exception e) {
        log.error("查询订单统计失败", queryParams, e);
        throw new BusinessException("统计查询失败");
    }
}
```

✅ 没有 `Map<String, Object>`
