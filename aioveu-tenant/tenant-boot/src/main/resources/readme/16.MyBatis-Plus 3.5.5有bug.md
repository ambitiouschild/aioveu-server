@Component
public class FixedTenantLineHandler implements TenantLineHandler {
    
    // 特殊标记：表示不添加租户条件
    private static final LongValue NO_TENANT_CONDITION = new LongValue(Long.MIN_VALUE);
    
    @Override
    public Expression getTenantId() {
        Long tenantId = TenantContextHolder.getTenantId();
        
        System.out.println("FixedTenantLineHandler - 租户ID: " + tenantId);
        
        // 场景1：登录等需要查询所有租户
        if (tenantId == null || tenantId == 0L) {
            System.out.println("返回特殊值 NO_TENANT_CONDITION");
            return NO_TENANT_CONDITION;
        }
        
        // 场景2：正常租户
        System.out.println("返回正常租户ID: " + tenantId);
        return new LongValue(tenantId);
    }
    
    @Override
    public String getTenantIdColumn() {
        return "tenant_id";
    }
    
    @Override
    public boolean ignoreTable(String tableName) {
        // 获取当前的租户ID表达式
        Long tenantId = TenantContextHolder.getTenantId();
        
        // 如果是特殊场景（登录等），忽略表
        if (tenantId == null || tenantId == 0L) {
            System.out.println("特殊场景，忽略表: " + tableName);
            return true;
        }
        
        // 系统表忽略
        if (tableName.startsWith("sys_")) {
            return true;
        }
        
        return false;
    }

### 方案3：自定义修复版本的 TenantLineInnerInterceptor

```
@Component
public class FixedTenantLineInnerInterceptor extends TenantLineInnerInterceptor {
    
    public FixedTenantLineInnerInterceptor(TenantLineHandler tenantLineHandler) {
        super(tenantLineHandler);
    }
    
    @Override
    protected Expression builderExpression(Expression where, List<Table> tables) {
        // 获取租户ID
        Expression tenantId = this.getTenantLineHandler().getTenantId();
        
        // 如果租户ID是null，直接返回原始where
        if (tenantId == null) {
            return where;
        }
        
        // 如果是特殊标记值，也返回原始where
        if (tenantId instanceof LongValue) {
            LongValue longValue = (LongValue) tenantId;
            if (longValue.getValue() == Long.MIN_VALUE) {
                return where;
            }
        }
        
        // 调用父类逻辑
        return super.builderExpression(where, tables);
    }
}
```

public class WorkaroundTenantLineHandler implements TenantLineHandler {

            return true;
        }
        
        // 系统表忽略
        return tableName.startsWith("sys_");
    }
    
    @Override
    public String getTenantIdColumn() {
        return "tenant_id";
    }
}
