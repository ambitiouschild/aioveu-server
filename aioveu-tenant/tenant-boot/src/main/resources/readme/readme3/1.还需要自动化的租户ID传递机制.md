**步骤1：在 User 实体上配置自动填充**

```
@Component
public class TenantIdMetaObjectHandler implements MetaObjectHandler {
    
    @Override
    public void insertFill(MetaObject metaObject) {
        // 只有 User 表才自动填充租户ID
        String tableName = metaObject.getOriginalObject().getClass().getSimpleName();
        if ("User".equalsIgnoreCase(tableName)) {
            // 从线程上下文获取租户ID
            Long tenantId = TenantContextHolder.getTenantId();
            if (tenantId != null) {
                this.strictInsertFill(metaObject, "tenantId", Long.class, tenantId);
                log.debug("自动填充用户表 tenant_id: {}", tenantId);
            }
        }
    }
    
    @Override
    public void updateFill(MetaObject metaObject) {
        // 更新时不修改租户ID
    }
}

**步骤3：配置 MyBatis-Plus**

```
@Configuration
public class MybatisPlusConfig {
    
    @Bean
    public MybatisPlusInterceptor mybatisPlusInterceptor() {
        MybatisPlusInterceptor interceptor = new MybatisPlusInterceptor();
        // 添加多租户插件
        interceptor.addInnerInterceptor(new TenantLineInnerInterceptor(
            new TenantLineHandler() {
                @Override
                public Expression getTenantId() {
                    Long tenantId = TenantContextHolder.getTenantId();
                    return tenantId != null ? new LongValue(tenantId) : null;
                }
                
                @Override
                public String getTenantIdColumn() {
                    return "tenant_id";
                }
                
                @Override
                public boolean ignoreTable(String tableName) {
                    // 这些表不需要租户过滤
                    return "sys_tenant".equalsIgnoreCase(tableName) 
                        || "sys_tenant_plan".equalsIgnoreCase(tableName);
                }
            }
        ));
        return interceptor;
    }
    
    @Bean
    public MetaObjectHandler metaObjectHandler(TenantIdMetaObjectHandler tenantIdMetaObjectHandler) {
        return tenantIdMetaObjectHandler;
    }
}

### 方案2：AOP 切面自动注入（无侵入）

**创建租户ID注入切面**：

@Aspect
@Component
@Slf4j
public class TenantIdInjectAspect {
    
    @Around("execution(* com..*Service.saveUser(..))")
    public Object injectTenantId(ProceedingJoinPoint joinPoint) throws Throwable {
        Object[] args = joinPoint.getArgs();
        
        for (int i = 0; i < args.length; i++) {
            if (args[i] instanceof UserForm) {
                UserForm userForm = (UserForm) args[i];
                
                // 如果用户没有设置tenantId，自动注入
                if (userForm.getTenantId() == null) {
                    Long currentTenantId = TenantContextHolder.getTenantId();
                    if (currentTenantId != null) {
                        userForm.setTenantId(currentTenantId);
                        log.debug("AOP自动注入租户ID: {}", currentTenantId);
                    }
                }
                break;
            }
        }
        
        return joinPoint.proceed(args);
    }
}

        // 自动填充租户ID

        if (tenantId != null && metaObject.hasSetter("tenantId")) {
            this.strictInsertFill(metaObject, "tenantId", Long.class, tenantId);
        }
        
        // 自动填充创建时间
        this.strictInsertFill(metaObject, "createTime", LocalDateTime.class, LocalDateTime.now());
        
        // 自动填充创建人

        }
    }

        // 自动填充更新时间

        // 自动填充更新人
