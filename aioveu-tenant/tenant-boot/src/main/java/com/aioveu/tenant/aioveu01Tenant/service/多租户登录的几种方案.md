## **后端 API 接口**

### **1. 获取租户列表接口**

```
@RestController
@RequestMapping("/api/v1/auth")
public class AuthController {
    
    @GetMapping("/tenants")
    public Result<List<TenantVO>> getTenantList(@RequestParam(required = false) String username) {
        // 如果有用户名，查询用户所属租户
        if (StringUtils.hasText(username)) {
            List<TenantVO> userTenants = tenantService.getTenantsByUsername(username);
            return Result.success(userTenants);
        }
        
        // 否则返回所有启用的租户
        List<TenantVO> allTenants = tenantService.listEnabledTenants();
        return Result.success(allTenants);
    }
    
    @PostMapping("/login")
    public Result<LoginResult> login(@RequestBody @Valid LoginForm form) {
        // 设置租户上下文
        if (form.getTenantId() != null) {
            TenantContext.setCurrentTenantId(form.getTenantId());
        }
        
        // 验证用户登录
        Authentication authentication = authenticationManager.authenticate(
            new UsernamePasswordAuthenticationToken(
                form.getUsername(),
                form.getPassword()
            )
        );
        
        LoginUser loginUser = (LoginUser) authentication.getPrincipal();
        
        // 生成token，包含租户ID
        String token = tokenService.createToken(loginUser);
        
        return Result.success(LoginResult.builder()
            .accessToken(token)
            .tenantId(form.getTenantId())
            .userInfo(UserVO.from(loginUser.getUser()))
            .build());
    }
}
```

import request from "@/utils/request";

export function getTenantList(username?: string) {

    params: { username },

// 登录
export function login(data: {
  tenantId?: number;
  username: string;
  password: string;

}
