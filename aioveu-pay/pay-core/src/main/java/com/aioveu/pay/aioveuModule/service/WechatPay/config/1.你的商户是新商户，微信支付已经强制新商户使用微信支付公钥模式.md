### 步骤3：创建正确的配置类

微信SDK的 `RSAAutoCertificateConfig`确实不支持公钥参数，但你可以用 `RSAConfig`：

```
@Service
@Slf4j
public class WeChatPayV3PublicKeyService {
    
    /**
     * 使用微信支付公钥创建配置
     */
    public com.wechat.pay.java.core.Config createConfig() {
        try {
            log.info("【微信支付】使用公钥模式创建配置");
            
            // 1. 加载私钥
            String privateKey = weChatPayConfig.getPrivateKey();
            
            // 2. 创建证书提供者
            CertificateProvider certificateProvider = createCertificateProvider();
            
            // 3. 创建RSA配置
            return new com.wechat.pay.java.core.RSAConfig.Builder()
                .merchantId(weChatPayConfig.getMchId())
                .privateKey(privateKey)
                .merchantSerialNumber(weChatPayConfig.getMchSerialNo())
                .certificateProvider(certificateProvider)
                .build();
                
        } catch (Exception e) {
            log.error("创建公钥模式配置失败", e);
            throw new RuntimeException("请确认已配置微信支付公钥", e);
        }
    }
    
    private CertificateProvider createCertificateProvider() {
        // 从你的配置文件获取
        String publicKeyId = weChatPayConfig.getWechatpayPublicKeyId();
        String publicKey = weChatPayConfig.getWechatpayPublicKey();
        
        if (StringUtils.isBlank(publicKeyId) || StringUtils.isBlank(publicKey)) {
            throw new IllegalArgumentException("请配置微信支付公钥和公钥ID");
        }
        
        // 创建公钥证书提供者
        return new com.wechat.pay.java.core.certificate.RSAPublicKeyCertificateProvider(
            publicKeyId, 
            publicKey
        );
    }
}
```
