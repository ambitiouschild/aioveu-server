```java
package com.aioveu.pay.aioveu10MqSendRecord.entity;

import lombok.Data;
import javax.persistence.*;
import java.time.LocalDateTime;

@Data
@Entity
@Table(name = "mq_send_record")
public class MqSendRecordEntity {
    
    // ========== 主键 ==========
    @Id
    @Column(name = "id", length = 32)
    private String id;
    
    // ========== 消息标识 ==========
    @Column(name = "message_id", length = 64, nullable = false)
    private String messageId;
    
    @Column(name = "correlation_id", length = 64)
    private String correlationId;
    
    // ========== 状态信息 ==========
    @Column(name = "send_status", nullable = false)
    private Integer sendStatus;
    
    @Column(name = "error_code", length = 50)
    private String errorCode;
    
    @Column(name = "error_msg", length = 500)
    private String errorMsg;
    
    // ========== 路由信息 ==========
    @Column(name = "exchange", length = 100, nullable = false)
    private String exchange;
    
    @Column(name = "routing_key", length = 200, nullable = false)
    private String routingKey;
    
    // ========== 业务信息 ==========
    @Column(name = "tenant_id", length = 50)
    private String tenantId;
    
    @Column(name = "message_type", length = 50, nullable = false)
    private String messageType;
    
    @Column(name = "business_id", length = 100)
    private String businessId;
    
    // ========== 消息内容 ==========
    @Column(name = "message_body", columnDefinition = "TEXT")
    private String messageBody;
    
    @Column(name = "message_size")
    private Integer messageSize;
    
    @Column(name = "content_type", length = 50)
    private String contentType = "application/json";
    
    // ========== 统计信息 ==========
    @Column(name = "cost_time")
    private Long costTime;
    
    @Column(name = "retry_count")
    private Integer retryCount = 0;
    
    @Column(name = "max_retry_count")
    private Integer maxRetryCount = 3;
    
    @Column(name = "retried")
    private Boolean retried = false;
    
    // ========== 时间信息 ==========
    @Column(name = "send_time", nullable = false)
    private LocalDateTime sendTime;
    
    @Column(name = "confirm_time")
    private LocalDateTime confirmTime;
    
    @Column(name = "expire_time")
    private LocalDateTime expireTime;
    
    @Column(name = "next_retry_time")
    private LocalDateTime nextRetryTime;
    
    @Column(name = "create_time", nullable = false, updatable = false)
    private LocalDateTime createTime = LocalDateTime.now();
    
    @Column(name = "update_time", nullable = false)
    private LocalDateTime updateTime = LocalDateTime.now();
    
    // ========== 扩展信息 ==========
    @Column(name = "extra_info", columnDefinition = "TEXT")
    private String extraInfo;
    
    // ========== 业务方法 ==========
    
    /**
     * 获取枚举状态
     */
    public SendStatus getSendStatusEnum() {
        return SendStatus.fromValue(this.sendStatus);
    }
    
    /**
     * 设置枚举状态
     */
    public void setSendStatusEnum(SendStatus status) {
        this.sendStatus = status != null ? status.getValue() : null;
    }
    
    /**
     * 判断是否成功
     */
    public boolean isSuccess() {
        return SendStatus.SUCCESS.getValue().equals(this.sendStatus);
    }
    
    /**
     * 判断是否可以重试
     */
    public boolean canRetry() {
        SendStatus status = getSendStatusEnum();
        boolean isRetryableStatus = status == SendStatus.FAILED || 
                                   status == SendStatus.TIMEOUT || 
                                   status == SendStatus.CONFIRM_TIMEOUT ||
                                   status == SendStatus.CONFIRM_NACK;
        
        boolean withinMaxRetry = this.retryCount < this.maxRetryCount;
        boolean notExpired = this.expireTime == null || 
                            LocalDateTime.now().isBefore(this.expireTime);
        boolean canRetryNow = this.nextRetryTime == null || 
                             LocalDateTime.now().isAfter(this.nextRetryTime);
        
        return isRetryableStatus && withinMaxRetry && notExpired && canRetryNow;
    }
    
    /**
     * 计算下次重试时间
     */
    public LocalDateTime calculateNextRetryTime() {
        int retryCount = this.retryCount != null ? this.retryCount : 0;
        // 指数退避：5秒, 10秒, 20秒, 40秒, 80秒, 最大300秒
        long delaySeconds = Math.min((long) Math.pow(2, retryCount) * 5, 300);
        return LocalDateTime.now().plusSeconds(delaySeconds);
    }
}
```

```java
package com.aioveu.pay.aioveu12MqProducerPayment.service;

import lombok.RequiredArgsConstructor;
import lombok.extern.slf4j.Slf4j;
import org.springframework.amqp.rabbit.core.RabbitTemplate;
import org.springframework.amqp.rabbit.support.CorrelationData;
import org.springframework.stereotype.Service;
import org.springframework.transaction.annotation.Transactional;
import java.time.LocalDateTime;
import java.util.UUID;
import java.util.concurrent.TimeUnit;

@Slf4j
@Service
@RequiredArgsConstructor
public class CompleteMessageSendService {
    
    private final MqSendRecordRepository mqSendRecordRepository;
    private final RabbitTemplate rabbitTemplate;
    
    /**
     * 完整的消息发送方法
     */
    @Transactional
    public RabbitSendResult sendMessage(CompleteMessageRequest request) {
        long startTime = System.currentTimeMillis();
        String messageId = generateMessageId();
        
        try {
            // 1. 创建并保存消息记录
            MqSendRecordEntity record = createMessageRecord(request, messageId, startTime);
            mqSendRecordRepository.save(record);
            
            // 2. 发送消息到RabbitMQ
            org.springframework.amqp.core.Message message = buildAmqpMessage(request, messageId);
            CorrelationData correlationData = new CorrelationData(messageId);
            
            rabbitTemplate.send(
                request.getExchange(),
                request.getRoutingKey(),
                message,
                correlationData
            );
            
            // 3. 等待确认
            CorrelationData.Confirm confirm = correlationData.getFuture()
                .get(request.getTimeoutMs(), TimeUnit.MILLISECONDS);
            
            long costTime = System.currentTimeMillis() - startTime;
            
            // 4. 更新记录并返回结果
            return handleSendResult(record, confirm, costTime, request);
            
        } catch (Exception e) {
            return handleSendException(messageId, request, e, startTime);
        }
    }
    
    /**
     * 创建消息记录（包含所有字段）
     */
    private MqSendRecordEntity createMessageRecord(CompleteMessageRequest request, 
                                                  String messageId, 
                                                  long startTime) {
        MqSendRecordEntity record = new MqSendRecordEntity();
        
        // 主键
        record.setId(generateId());
        
        // 消息标识
        record.setMessageId(messageId);
        record.setCorrelationId(request.getCorrelationId() != null ? 
            request.getCorrelationId() : generateCorrelationId());
        
        // 状态
        record.setSendStatusEnum(SendStatus.SENDING);
        
        // 路由信息
        record.setExchange(request.getExchange());
        record.setRoutingKey(request.getRoutingKey());
        
        // 业务信息
        record.setTenantId(request.getTenantId());
        record.setMessageType(request.getMessageType());
        record.setBusinessId(request.getBusinessId());
        
        // 消息内容
        record.setMessageBody(request.getMessageBody());
        if (request.getMessageBody() != null) {
            record.setMessageSize(request.getMessageBody().getBytes().length);
        }
        record.setContentType(request.getContentType());
        
        // 重试配置
        record.setMaxRetryCount(request.getMaxRetryCount());
        if (request.getExpireSeconds() != null && request.getExpireSeconds() > 0) {
            record.setExpireTime(LocalDateTime.now().plusSeconds(request.getExpireSeconds()));
        }
        
        // 时间
        record.setSendTime(LocalDateTime.now());
        
        return record;
    }
    
    /**
     * 处理发送结果
     */
    private RabbitSendResult handleSendResult(MqSendRecordEntity record,
                                             CorrelationData.Confirm confirm,
                                             long costTime,
                                             CompleteMessageRequest request) {
        if (confirm.isAck()) {
            // 成功
            record.setSendStatusEnum(SendStatus.SUCCESS);
            record.setConfirmTime(LocalDateTime.now());
            record.setCostTime(costTime);
            mqSendRecordRepository.save(record);
            
            return RabbitSendResult.success(
                record.getMessageId(),
                record.getCorrelationId(),
                costTime,
                record.getExchange(),
                record.getRoutingKey()
            )
            .withTenant(record.getTenantId())
            .withMessageType(record.getMessageType())
            .addExtraInfo("businessId", record.getBusinessId())
            .addExtraInfo("messageSize", record.getMessageSize());
            
        } else {
            // Broker返回NACK
            record.setSendStatusEnum(SendStatus.CONFIRM_NACK);
            record.setErrorMsg("Broker返回NACK: " + confirm.getReason());
            record.setCostTime(costTime);
            mqSendRecordRepository.save(record);
            
            return RabbitSendResult.failure(
                record.getMessageId(),
                "Broker返回NACK: " + confirm.getReason(),
                costTime,
                record.getExchange(),
                record.getRoutingKey()
            )
            .withTenant(record.getTenantId())
            .withMessageType(record.getMessageType());
        }
    }
    
    // ... 其他方法
}
```

```java
package com.aioveu.pay.aioveu12MqProducerPayment.service;

@Slf4j
@Service
@RequiredArgsConstructor
public class CompleteMessageRetryService {
    
    private final MqSendRecordRepository mqSendRecordRepository;
    private final RabbitTemplate rabbitTemplate;
    
    /**
     * 重试消息（使用所有字段）
     */
    @Transactional
    public RetryResult retryMessage(String messageId) {
        long startTime = System.currentTimeMillis();
        
        try {
            // 1. 查询消息
            MqSendRecordEntity entity = mqSendRecordRepository.findByMessageId(messageId)
                .orElseThrow(() -> new IllegalArgumentException("消息不存在: " + messageId));
            
            // 2. 检查是否可以重试
            if (!entity.canRetry()) {
                return createCannotRetryResult(entity, startTime);
            }
            
            // 3. 更新为重试中
            entity.setSendStatusEnum(SendStatus.SENDING);
            entity.setRetryCount(entity.getRetryCount() + 1);
            entity.setRetried(true);
            entity.setNextRetryTime(entity.calculateNextRetryTime());
            entity.setUpdateTime(LocalDateTime.now());
            mqSendRecordRepository.save(entity);
            
            // 4. 执行重试
            RabbitSendResult rabbitResult = executeRetry(entity, startTime);
            
            // 5. 更新结果
            return updateAfterRetry(entity, rabbitResult, startTime);
            
        } catch (Exception e) {
            return createExceptionResult(messageId, e, startTime);
        }
    }
    
    /**
     * 执行重试发送
     */
    private RabbitSendResult executeRetry(MqSendRecordEntity entity, long startTime) {
        // 使用实体中的所有字段
        String exchange = entity.getExchange();
        String routingKey = entity.getRoutingKey();
        String correlationId = entity.getCorrelationId();
        
        try {
            // 构建消息
            org.springframework.amqp.core.Message message = buildRetryMessage(entity);
            CorrelationData correlationData = new CorrelationData(
                correlationId != null ? correlationId : entity.getMessageId()
            );
            
            // 发送
            rabbitTemplate.send(exchange, routingKey, message, correlationData);
            
            // 等待确认
            CorrelationData.Confirm confirm = correlationData.getFuture()
                .get(5000, TimeUnit.MILLISECONDS);
            
            long costTime = System.currentTimeMillis() - startTime;
            
            if (confirm.isAck()) {
                return RabbitSendResult.success(
                    entity.getMessageId(),
                    correlationData.getId(),
                    costTime,
                    exchange,
                    routingKey
                );
            } else {
                return RabbitSendResult.failure(
                    entity.getMessageId(),
                    "Broker返回NACK: " + confirm.getReason(),
                    costTime,
                    exchange,
                    routingKey
                );
            }
            
        } catch (TimeoutException e) {
            long costTime = System.currentTimeMillis() - startTime;
            return RabbitSendResult.timeout(
                entity.getMessageId(),
                costTime,
                exchange,
                routingKey
            );
        } catch (Exception e) {
            long costTime = System.currentTimeMillis() - startTime;
            return RabbitSendResult.failure(
                entity.getMessageId(),
                e.getMessage(),
                costTime,
                exchange,
                routingKey
            );
        }
    }
}
```
