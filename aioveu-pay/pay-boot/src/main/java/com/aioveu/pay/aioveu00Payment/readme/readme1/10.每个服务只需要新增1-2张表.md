```
// 1. 发送记录实体
@Data
@TableName("mq_send_record")
public class MqSendRecord {
    private Long id;
    private String messageId;
    private String bizId;        // payment_no
    private String topic;        // "payment_success_topic"
    private String tag;          // "wechat_pay"
    private Integer sendStatusEnum;  // 0-未发送,1-发送成功,2-发送失败
    private String errorMsg;
    private Date createTime;
}

// 2. Mapper
@Mapper
public interface MqSendRecordMapper extends BaseMapper<MqSendRecord> {
    
    @Select("SELECT * FROM mq_send_record WHERE biz_id = #{bizId} AND topic = #{topic} ORDER BY id DESC LIMIT 1")
    MqSendRecord selectLatestByBizId(@Param("bizId") String bizId, @Param("topic") String topic);
    
    @Update("UPDATE mq_send_record SET send_status = #{status}, error_msg = #{errorMsg} WHERE message_id = #{messageId}")
    int updateStatus(@Param("messageId") String messageId, 
                    @Param("status") Integer status, 
                    @Param("errorMsg") String errorMsg);
}

// 3. 发送服务
@Service
@Slf4j
public class PaymentMQService {
    
    @Autowired
    private MqSendRecordMapper sendRecordMapper;
    
    @Autowired
    private RocketMQTemplate rocketMQTemplate;
    
    /**
     * 发送支付成功消息
     */
    public boolean sendPaymentSuccess(PayOrder payOrder, Map<String, String> wechatParams) {
        String messageId = UUID.randomUUID().toString();
        
        try {
            // 1. 保存发送记录
            MqSendRecord record = new MqSendRecord();
            record.setMessageId(messageId);
            record.setBizId(payOrder.getPaymentNo());
            record.setTopic("payment_success_topic");
            record.setTag("wechat_pay");
            record.setSendStatus(0); // 未发送
            sendRecordMapper.insert(record);
            
            // 2. 构建消息
            PaymentSuccessMessage message = PaymentSuccessMessage.builder()
                .messageId(messageId)
                .paymentNo(payOrder.getPaymentNo())
                .orderNo(payOrder.getOrderNo())
                .transactionId(wechatParams.get("transaction_id"))
                .amount(payOrder.getAmount())
                .build();
            
            // 3. 发送消息
            SendResult result = rocketMQTemplate.syncSendOrderly(
                "payment_success_topic:wechat_pay",
                MessageBuilder.withPayload(message).build(),
                payOrder.getOrderNo(),
                3000
            );
            
            // 4. 更新发送状态
            if (result.getSendStatus() == SendStatus.SEND_OK) {
                sendRecordMapper.updateStatus(messageId, 1, null);
                return true;
            } else {
                sendRecordMapper.updateStatus(messageId, 2, "发送失败");
                return false;
            }
            
        } catch (Exception e) {
            log.error("发送MQ消息失败: messageId={}", messageId, e);
            sendRecordMapper.updateStatus(messageId, 2, e.getMessage());
            return false;
        }
    }
}
```

```
// 3. 幂等性检查服务
@Service
@Slf4j
public class MqIdempotentService {
    
    @Autowired
    private MqConsumeRecordMapper consumeRecordMapper;
    
    /**
     * 检查消息是否已消费
     */
    public boolean isMessageConsumed(String messageId, String topic) {
        MqConsumeRecord record = consumeRecordMapper.selectByMessage(messageId, topic);
        return record != null && record.getConsumeStatus() == 1;
    }
    
    /**
     * 记录消费
     */
    public void recordConsume(String messageId, String topic, String orderNo, boolean success, String errorMsg) {
        MqConsumeRecord record = new MqConsumeRecord();
        record.setMessageId(messageId);
        record.setTopic(topic);
        record.setBizId(orderNo);
        record.setConsumeStatus(success ? 1 : 2);
        record.setErrorMsg(errorMsg);
        consumeRecordMapper.insert(record);
    }
}
```

### 方案1：使用Redis记录状态

```
// 支付服务 - 发送状态存Redis
@Component
public class PaymentMQSender {
    
    @Autowired
    private RedisTemplate<String, String> redisTemplate;
    
    public void sendMessage(PayOrder payOrder) {
        String key = "mq:send:" + payOrder.getPaymentNo();
        
        // 检查是否已发送
        if (Boolean.TRUE.equals(redisTemplate.hasKey(key))) {
            log.warn("消息已发送，跳过: {}", payOrder.getPaymentNo());
            return;
        }
        
        // 发送消息
        rocketMQTemplate.send(...);
        
        // 记录发送状态，24小时过期
        redisTemplate.opsForValue().set(key, "1", 24, TimeUnit.HOURS);
    }
}

@Component
public class OrderMQConsumer {

        // 检查是否已消费
        if (Boolean.TRUE.equals(redisTemplate.hasKey(key))) {
            return; // 已消费
        }
        
        // 处理业务
        orderService.handlePaymentSuccess(...);
        
        // 记录已消费，24小时过期
        redisTemplate.opsForValue().set(key, "1", 24, TimeUnit.HOURS);
    }
}
```

## 🎯 最终建议

**建议采用最小化方案**：

1. **支付服务**：新增 `mq_send_record`表
2. **订单服务**：新增 `mq_consume_record`表
3. **先实现核心功能**，补偿表和死信表可以后续再加

**优势**：

- 改动最小
- 满足核心需求
- 扩展性好
- 易于排查问题

两张表的字段可以保持简单，后面根据需求再扩展。
