```
@Override
public String handleWechatCallback(String xmlData) {

    long start = System.currentTimeMillis();

    try {
        // 1. 解析
        Map<String, String> params = parseWechatXml(xmlData);

        // 2. 验签
        if (!verifyWechatSign(params)) {
            log.error("【微信回调】签名失败");
            return success(); // ✅ 防止微信重试
        }

        // 3. 基础校验
        String transactionId = params.get("transaction_id");
        String paymentNo = params.get("out_trade_no");

        if (!"SUCCESS".equals(params.get("return_code")) ||
            !"SUCCESS".equals(params.get("result_code"))) {
            log.warn("【微信回调】支付未成功, paymentNo={}", paymentNo);
            return success();
        }

        // 4. ✅【核心】幂等判断（必须用 transaction_id）
        if (mqConsumeRecordService.isConsumed(transactionId)) {
            log.info("【微信回调】已处理，直接返回 SUCCESS, transactionId={}", transactionId);
            return success();
        }

        // 5. 查询支付单
        PayOrder payOrder = payOrderService.getByPaymentNo(paymentNo);
        if (payOrder == null) {
            log.error("【微信回调】支付单不存在, paymentNo={}", paymentNo);
            return success();
        }

        // 6. ✅ 业务处理（事务内）
        processWechatPaySuccess(payOrder, params, transactionId);

        log.info("【微信回调】处理完成, cost={}ms", System.currentTimeMillis() - start);
        return success();

    } catch (Exception e) {
        log.error("【微信回调】系统异常", e);
        return success(); // ✅ 永远不给微信 FAIL
    }
}
```

# ✅ 事务方法（✅ 核心）

```
@Transactional(rollbackFor = Exception.class)
public void processWechatPaySuccess(
        PayOrder payOrder,
        Map<String, String> params,
        String transactionId) {

    // 1. 状态校验
    if (!isProcessable(payOrder)) {
        log.warn("【微信回调】订单状态不可处理, paymentNo={}, status={}",
                payOrder.getPaymentNo(), payOrder.getPaymentStatus());
        return;
    }

    // 2. 金额校验
    if (!verifyAmount(payOrder, params)) {
        throw new BizException("金额不匹配");
    }

    // 3. 更新支付单
    payOrderService.updatePaymentStatus(payOrder, params, transactionId);

    // 4. ✅ 幂等落库（必须在事务内）
    mqConsumeRecordService.markConsumed(
            transactionId,
            payOrder.getPaymentNo(),
            payOrder.getOrderNo()
    );

    // 5. 发送 MQ
    SendPaymentMqDTO dto = buildPaymentMqDTO(payOrder, params, transactionId);
    boolean mqSuccess = payCommonMessageProducerService.sendPaymentSuccessMessage(dto);

    // 6. MQ 失败 → 补偿
    if (!mqSuccess) {
        saveToCompensation(payOrder, params, "MQ_SEND_FAILED");
    }

    // 7. 流水
    payFlowService.recordPaymentFlow(payOrder, params);
}
```

```
private SendPaymentMqDTO buildPaymentMqDTO(
        PayOrder payOrder,
        Map<String, String> params,
        String transactionId) {

    SendPaymentMqDTO dto = new SendPaymentMqDTO();
    dto.setMessageId(messageIdGenerator.generatePaymentMessageId(payOrder.getPaymentNo()));
    dto.setPaymentNo(payOrder.getPaymentNo());
    dto.setOmsOrderNo(payOrder.getOrderNo());
    dto.setPaymentAmount(payOrder.getPaymentAmount());
    dto.setTransactionId(transactionId);
    dto.setPaymentTime(LocalDateTime.now());
    dto.setBizTypeEnum(PaymentMqBizType.PAYMENT_SUCCESS);
    return dto;
}
```
