@Slf4j

public class PaymentMessageController {

# ✅ 三、Service 层（你真正写业务逻辑的地方）

```
public interface PaymentMessageService {

    void sendPaymentSuccessMessage(SendPaymentMqDTO dto);

    void sendPaymentFailedMessage(SendPaymentMqDTO dto);
}
```

------

```
@Slf4j
@Service
@RequiredArgsConstructor
public class PaymentMessageServiceImpl implements PaymentMessageService {

    private final RocketMQTemplate rocketMQTemplate;
    private final PayOrderMapper payOrderMapper;
    private final MqSendRecordMapper mqSendRecordMapper;
    private final ObjectMapper objectMapper;

    @Override
    @Transactional
    public void sendPaymentSuccessMessage(SendPaymentMqDTO dto) {

        PayOrder payOrder = payOrderMapper.selectByPayOrderNo(dto.getPayOrderNo());
        if (payOrder == null) {
            throw new BusinessException("支付订单不存在");
        }

        PaymentSuccessMessage message = buildMessage(payOrder, dto);

        // 1️⃣ 保存发送记录
        saveSendRecord(message, payOrder);

        // 2️⃣ 发送 MQ
        rocketMQTemplate.convertAndSend(
                "payment_success_topic",
                message
        );

        log.info("支付成功消息已发送: {}", message.getMessageId());
    }

    @Override
    @Transactional
    public void sendPaymentFailedMessage(SendPaymentMqDTO dto) {

        PayOrder payOrder = payOrderMapper.selectByPayOrderNo(dto.getPayOrderNo());
        if (payOrder == null) {
            throw new BusinessException("支付订单不存在");
        }

        PaymentFailedMessage message = buildFailedMessage(payOrder, dto);

        saveSendRecord(message, payOrder);

        rocketMQTemplate.convertAndSend(
                "payment_failed_topic",
                message
        );
    }

    private PaymentSuccessMessage buildMessage(PayOrder payOrder, SendPaymentMqDTO dto) {
        PaymentSuccessMessage msg = new PaymentSuccessMessage();
        msg.setMessageId(UUID.randomUUID().toString());
        msg.setOrderNo(payOrder.getOrderNo());
        msg.setTenantId(dto.getTenantId());
        msg.setPaymentTime(OffsetDateTime.now());
        msg.setPaidAmount(payOrder.getAmount());
        return msg;
    }

    private void saveSendRecord(Object message, PayOrder payOrder) {
        MqSendRecord record = new MqSendRecord();
        record.setMessageId(UUID.randomUUID().toString());
        record.setTopic("payment_success_topic");
        record.setBizKey(payOrder.getOrderNo());
        record.setStatus(MqSendStatusEnum.SENT.getValue());
        record.setCreateTime(LocalDateTime.now());
        mqSendRecordMapper.insert(record);
    }
}
```
