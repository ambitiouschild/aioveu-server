## 2️⃣ PaymentStatusEnum（状态机核心）

```
public enum PaymentStatusEnum {

    PENDING(0, "待支付"),
    PAID(1, "已支付"),
    CLOSED(2, "已关闭");

    private final int code;
    private final String desc;

    public static PaymentStatusEnum fromCode(Integer code) {
        for (PaymentStatusEnum e : values()) {
            if (e.code == code) return e;
        }
        return null;
    }

    /** 是否终态 */
    public boolean isTerminal() {
        return this == PAID || this == CLOSED;
    }
}
```

<select id="selectPendingOrdersForQuery">

      AND created_at &gt;= #{maxCreatedAt}
      AND created_at &lt;= #{lastCreatedAt}
      AND id &lt; #{lastId}
    ORDER BY created_at DESC, id DESC
    LIMIT #{limit}

<update id="updateStatusIfNonTerminal">
    UPDATE pay_order
    SET payment_status = #{status},
        third_transaction_no = #{thirdNo},
        payment_time = #{paymentTime},
        last_query_time = NOW(),
        updated_at = NOW(),
        version = version + 1
    WHERE id = #{id}
      AND payment_status IN (0, 10)
</update>

<update id="updateLastQueryTime">
    UPDATE pay_order
    SET last_query_time = #{time},
        updated_at = NOW()
    WHERE id = #{id}

# 四、PaymentRecoveryService（核心兜底逻辑）

```
@Service
@Slf4j
public class PaymentRecoveryService {

    @Resource
    private PayOrderMapper payOrderMapper;
    @Resource
    private WechatPayService wechatPayService;
    @Resource
    private BusinessProcessor businessProcessor;

    /**
     * 单笔订单兜底查单
     */
    @Transactional(rollbackFor = Exception.class)
    public void recover(String paymentNo) {
        PayOrder order = payOrderMapper.selectOne(
                Wrappers.<PayOrder>lambdaQuery()
                        .eq(PayOrder::getPaymentNo, paymentNo)
        );

        if (order == null || order.getPaymentStatus().isTerminal()) {
            return;
        }

        // 5 分钟内查过，直接跳过
        if (order.getLastQueryTime() != null &&
            order.getLastQueryTime().isAfter(LocalDateTime.now().minusMinutes(5))) {
            return;
        }

        try {
            PaymentStatusVO wx = wechatPayService.queryPayment(paymentNo);

            if (wx == null || !wx.getPaymentStatus().isTerminal()) {
                payOrderMapper.updateLastQueryTime(
                        order.getId(),
                        LocalDateTime.now()
                );
                return;
            }

            int rows = payOrderMapper.updateStatusIfNonTerminal(
                    order.getId(),
                    wx.getPaymentStatus().getCode(),
                    wx.getThirdPaymentNo(),
                    wx.getPaymentTime()
            );

            if (rows == 1) {
                log.info("兜底查单成功, paymentNo={}", paymentNo);
                businessProcessor.onPaid(paymentNo);
            }

        } catch (Exception e) {
            log.error("兜底查单异常, paymentNo={}", paymentNo, e);
        }
    }
}
```

# 五、定时任务（Job）——第三层兜底入口

```
@Component
@Slf4j
public class PaymentReconciliationJob {

    @Resource
    private PayOrderMapper payOrderMapper;
    @Resource
    private PaymentRecoveryService recoveryService;

    /**
     * 每分钟执行一次
     */
    @Scheduled(cron = "0 */1 * * * ?")
    public void reconcile() {
        log.info("支付兜底Job开始执行");

        LocalDateTime maxCreatedAt = LocalDateTime.now().minusMinutes(30);
        LocalDateTime lastCreatedAt = LocalDateTime.now();
        Long lastId = Long.MAX_VALUE;
        int batchSize = 200;

        while (true) {
            List<PayOrder> orders = payOrderMapper
                    .selectPendingOrdersForQuery(
                            maxCreatedAt,
                            lastCreatedAt,
                            lastId,
                            batchSize
                    );

            if (CollUtil.isEmpty(orders)) {
                break;
            }

            orders.forEach(o -> recoveryService.recover(o.getPaymentNo()));

            PayOrder last = orders.get(orders.size() - 1);
            lastId = last.getId();
            lastCreatedAt = last.getCreatedAt();
        }

        log.info("支付兜底Job执行结束");
    }
}
```
