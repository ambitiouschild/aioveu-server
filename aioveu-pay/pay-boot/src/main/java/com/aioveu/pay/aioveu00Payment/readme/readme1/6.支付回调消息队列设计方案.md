# 🎯 支付回调消息队列设计方案

## 📊 整体架构
```
┌─────────────────────────────────────────────────────────────┐
│                    支付回调处理流程                           │
├─────────────────────────────────────────────────────────────┤
│ 1. 微信回调 → 支付服务                                       │
│ 2. 支付服务: 验证签名、更新支付订单、发送MQ                   │
│ 3. MQ: 消息路由、持久化、重试                                │
│ 4. 消费者集群: 订单服务、积分服务、通知服务等                 │
│ 5. 监控告警: 监控面板、告警系统                              │
└─────────────────────────────────────────────────────────────┘
```

## 1. 技术选型

### 消息队列：**RocketMQ**
• ✅ 顺序消息（同一订单顺序处理）

• ✅ 事务消息（保证本地事务和消息发送一致）

• ✅ 高可用（主从架构）

• ✅ 丰富的监控指标

• ✅ 阿里巴巴出品，电商场景经验丰富


### 备选：Kafka
• 如果已有Kafka集群可以使用

• 但顺序消息和事务消息支持不如RocketMQ

## 4. 核心代码实现

### 4.1 支付服务 - 回调处理
```java
@Service
@Slf4j
@RequiredArgsConstructor
public class PaymentCallbackHandler {
    
    private final PayOrderService payOrderService;
    private final MQMessageService mqMessageService;
    private final RocketMQTemplate rocketMQTemplate;
    
    /**
     * 处理微信支付回调
     */
    public String handleWechatCallback(String xmlData) {
        String paymentNo = null;
        long startTime = System.currentTimeMillis();
        
        try {
            log.info("【微信回调】开始处理, XML长度: {}", xmlData.length());
            
            // 1. 解析XML
            Map<String, String> params = parseWechatXml(xmlData);
            paymentNo = params.get("out_trade_no");
            
            // 2. 验证签名
            if (!verifyWechatSign(params)) {
                log.error("【微信回调】签名验证失败: paymentNo={}", paymentNo);
                return generateWechatFailResponse("签名验证失败");
            }
            
            // 3. 验证业务返回码
            String returnCode = params.get("return_code");
            String resultCode = params.get("result_code");
            
            if (!"SUCCESS".equals(returnCode)) {
                log.error("【微信回调】通信失败: paymentNo={}, msg={}", 
                    paymentNo, params.get("return_msg"));
                return generateWechatFailResponse(params.get("return_msg"));
            }
            
            // 4. 查询支付订单
            PayOrder payOrder = payOrderService.getByPaymentNo(paymentNo);
            if (payOrder == null) {
                log.error("【微信回调】支付订单不存在: paymentNo={}", paymentNo);
                return generateWechatFailResponse("订单不存在");
            }
            
            // 5. 幂等性检查
            if (payOrder.getPaymentStatus() != PaymentStatusEnum.PENDING.getValue() 
                && payOrder.getPaymentStatus() != PaymentStatusEnum.PROCESSING.getValue()) {
                log.warn("【微信回调】订单已处理，跳过: paymentNo={}, status={}", 
                    paymentNo, payOrder.getPaymentStatus());
                return generateWechatSuccessResponse();
            }
            
            // 6. 验证金额
            BigDecimal callbackAmount = new BigDecimal(params.get("total_fee"))
                .divide(new BigDecimal(100), 2, RoundingMode.HALF_UP);
            if (callbackAmount.compareTo(payOrder.getAmount()) != 0) {
                log.error("【微信回调】金额不匹配: paymentNo={}, 订单金额={}, 回调金额={}", 
                    paymentNo, payOrder.getAmount(), callbackAmount);
                return generateWechatFailResponse("金额不匹配");
            }
            
            // 7. 更新支付订单
            boolean success = "SUCCESS".equals(resultCode);
            payOrderService.updatePaymentStatus(payOrder, success, params);
            
            // 8. 发送MQ消息
            if (success) {
                sendPaymentSuccessMessage(payOrder, params);
            } else {
                sendPaymentFailedMessage(payOrder, params);
            }
            
            // 9. 记录处理时间
            long costTime = System.currentTimeMillis() - startTime;
            log.info("【微信回调】处理完成: paymentNo={}, 耗时={}ms", paymentNo, costTime);
            
            return generateWechatSuccessResponse();
            
        } catch (Exception e) {
            log.error("【微信回调】处理异常: paymentNo={}", paymentNo, e);
            return generateWechatFailResponse("系统异常");
        }
    }
    
    /**
     * 发送支付成功消息
     */
    private void sendPaymentSuccessMessage(PayOrder payOrder, Map<String, String> params) {
        try {
            // 构建消息
            PaymentSuccessMessage message = buildPaymentSuccessMessage(payOrder, params);
            
            // 保存发送记录
            String messageId = mqMessageService.saveSendRecord(
                "payment_success_topic", 
                "wechat_pay", 
                payOrder.getPaymentNo(), 
                message
            );
            
            // 发送顺序消息（确保同一订单的顺序性）
            SendResult sendResult = rocketMQTemplate.syncSendOrderly(
                "payment_success_topic:wechat_pay",
                MessageBuilder
                    .withPayload(message)
                    .setHeader(MessageConst.PROPERTY_KEYS, payOrder.getOrderNo())
                    .build(),
                payOrder.getOrderNo(),  // 使用订单号作为shardingKey
                3000
            );
            
            // 更新发送状态
            mqMessageService.updateSendStatus(messageId, SendStatusEnum.SUCCESS, null);
            
            log.info("【MQ发送】支付成功消息发送成功: paymentNo={}, messageId={}", 
                payOrder.getPaymentNo(), messageId);
            
        } catch (Exception e) {
            log.error("【MQ发送】支付成功消息发送失败: paymentNo={}", 
                payOrder.getPaymentNo(), e);
            
            // 记录发送失败，后续补偿任务会重试
            mqMessageService.updateSendStatus(
                UUID.randomUUID().toString(), 
                SendStatusEnum.FAILED, 
                e.getMessage()
            );
        }
    }
}
```

### 4.3 MQ消息服务
```java
@Service
@Slf4j
@RequiredArgsConstructor
public class MQMessageServiceImpl implements MQMessageService {
    
    private final MQMessageRecordMapper messageRecordMapper;
    private final RocketMQTemplate rocketMQTemplate;
    
    @Override
    @Transactional(rollbackFor = Exception.class)
    public String saveSendRecord(String topic, String tag, String bizId, Object message) {
        String messageId = generateMessageId();
        
        MQMessageRecord record = new MQMessageRecord();
        record.setMessageId(messageId);
        record.setBizId(bizId);
        record.setTopic(topic);
        record.setTag(tag);
        record.setShardingKey(bizId);  // 使用业务ID作为分片key
        record.setMessageBody(JSON.toJSONString(message));
        record.setSendStatus(SendStatusEnum.PENDING.getValue());
        record.setCreateTime(new Date());
        record.setUpdateTime(new Date());
        
        messageRecordMapper.insert(record);
        
        return messageId;
    }
    
    @Override
    public void updateSendStatus(String messageId, SendStatusEnum status, String errorMsg) {
        MQMessageRecord record = new MQMessageRecord();
        record.setMessageId(messageId);
        record.setSendStatus(status.getValue());
        record.setErrorMsg(errorMsg);
        record.setUpdateTime(new Date());
        
        if (status == SendStatusEnum.SUCCESS) {
            record.setConfirmTime(new Date());
        } else if (status == SendStatusEnum.FAILED) {
            // 失败时设置下次重试时间（指数退避）
            MQMessageRecord oldRecord = messageRecordMapper.selectById(messageId);
            int retryCount = oldRecord.getRetryCount() + 1;
            long nextRetryTime = System.currentTimeMillis() + 
                (long) (Math.pow(2, retryCount) * 30000);  // 30秒 * 2^retryCount
            
            record.setRetryCount(retryCount);
            record.setNextRetryTime(new Date(nextRetryTime));
        }
        
        messageRecordMapper.updateById(record);
    }
    
    /**
     * 补偿任务 - 处理发送失败的消息
     */
    @Scheduled(fixedDelay = 30000)  // 30秒执行一次
    public void retryFailedMessages() {
        try {
            List<MQMessageRecord> failedRecords = messageRecordMapper.selectFailedMessages(100);
            
            for (MQMessageRecord record : failedRecords) {
                if (record.getNextRetryTime().after(new Date())) {
                    continue;  // 未到重试时间
                }
                
                if (record.getRetryCount() >= 5) {
                    log.error("消息重试超过5次，进入死信: messageId={}", record.getMessageId());
                    updateSendStatus(record.getMessageId(), SendStatusEnum.DEAD, "重试超过5次");
                    continue;
                }
                
                try {
                    // 重新发送消息
                    Message<byte[]> message = MessageBuilder
                        .withPayload(record.getMessageBody().getBytes(StandardCharsets.UTF_8))
                        .setHeader(MessageConst.PROPERTY_KEYS, record.getBizId())
                        .build();
                    
                    SendResult sendResult = rocketMQTemplate.syncSendOrderly(
                        record.getTopic() + ":" + record.getTag(),
                        message,
                        record.getShardingKey(),
                        3000
                    );
                    
                    if (sendResult.getSendStatus() == SendStatus.SEND_OK) {
                        updateSendStatus(record.getMessageId(), SendStatusEnum.SUCCESS, null);
                        log.info("补偿发送成功: messageId={}", record.getMessageId());
                    } else {
                        updateSendStatus(record.getMessageId(), SendStatusEnum.FAILED, "发送失败");
                    }
                    
                } catch (Exception e) {
                    log.error("补偿发送异常: messageId={}", record.getMessageId(), e);
                    updateSendStatus(record.getMessageId(), SendStatusEnum.FAILED, e.getMessage());
                }
            }
            
        } catch (Exception e) {
            log.error("重试失败消息异常", e);
        }
    }
}
```

### 4.4 订单服务 - 消费者
```java
@Service
@Slf4j
@RocketMQMessageListener(
    topic = "${rocketmq.topics.payment-success:payment_success_topic}",
    consumerGroup = "${rocketmq.consumer-groups.order-service:order_service_group}",
    selectorExpression = "wechat_pay || alipay",  // 只消费微信和支付宝支付消息
    consumeMode = ConsumeMode.ORDERLY,           // 顺序消费
    messageModel = MessageModel.CLUSTERING,      // 集群模式
    consumeThreadMax = 20,                       // 最大消费线程
    consumeTimeout = 15L,                        // 消费超时时间(分钟)
    suspendCurrentQueueTimeMillis = 1000,        // 消费失败挂起时间
    delayLevelWhenNextConsume = 0                // 消费失败延迟级别
)
public class PaymentSuccessConsumer implements RocketMQListener<MessageExt> {
    
    @Autowired
    private OrderService orderService;
    
    @Autowired
    private MQConsumeRecordService consumeRecordService;
    
    @Autowired
    private ObjectMapper objectMapper;
    
    @Override
    @Transactional(rollbackFor = Exception.class)
    public void onMessage(MessageExt messageExt) {
        String messageId = messageExt.getMsgId();
        String orderNo = messageExt.getKeys();  // 从keys获取订单号
        
        long startTime = System.currentTimeMillis();
        
        try {
            log.info("【订单消费】开始处理消息: messageId={}, orderNo={}", messageId, orderNo);
            
            // 1. 幂等性检查
            ConsumeStatusEnum consumeStatus = consumeRecordService.getConsumeStatus(messageId, "order_service_group");
            if (consumeStatus == ConsumeStatusEnum.SUCCESS) {
                log.warn("【订单消费】消息已消费，跳过: messageId={}", messageId);
                return;
            }
            
            // 2. 保存消费记录
            consumeRecordService.saveConsumeRecord(messageId, messageExt.getTopic(), 
                messageExt.getTags(), "order_service_group", orderNo);
            
            // 3. 解析消息
            PaymentSuccessMessage paymentMessage = objectMapper.readValue(
                messageExt.getBody(), 
                PaymentSuccessMessage.class
            );
            
            // 4. 查询订单
            Order order = orderService.getByOrderNo(orderNo);
            if (order == null) {
                log.error("【订单消费】订单不存在: orderNo={}", orderNo);
                consumeRecordService.updateConsumeStatus(messageId, "order_service_group", 
                    ConsumeStatusEnum.FAILED, "订单不存在");
                return;
            }
            
            // 5. 幂等性检查订单状态
            if (order.getStatus() != OrderStatusEnum.UNPAID.getCode()) {
                log.warn("【订单消费】订单状态不是待支付，跳过: orderNo={}, status={}", 
                    orderNo, order.getStatus());
                consumeRecordService.updateConsumeStatus(messageId, "order_service_group", 
                    ConsumeStatusEnum.SUCCESS, "订单已处理");
                return;
            }
            
            // 6. 更新订单状态
            boolean success = orderService.updateOrderPayStatus(
                orderNo,
                OrderStatusEnum.UNSHIPPED,  // 待发货
                paymentMessage.getTransactionId(),
                Date.from(paymentMessage.getPaymentTime().atZone(ZoneId.systemDefault()).toInstant()),
                paymentMessage.getAmount()
            );
            
            if (!success) {
                throw new RuntimeException("更新订单状态失败");
            }
            
            // 7. 记录操作日志
            orderService.recordPaymentSuccess(
                order.getId(),
                paymentMessage.getTransactionId(),
                paymentMessage.getAmount(),
                paymentMessage.getChannel()
            );
            
            // 8. 更新消费状态
            consumeRecordService.updateConsumeStatus(messageId, "order_service_group", 
                ConsumeStatusEnum.SUCCESS, null);
            
            long costTime = System.currentTimeMillis() - startTime;
            log.info("【订单消费】处理完成: orderNo={}, 耗时={}ms", orderNo, costTime);
            
        } catch (Exception e) {
            log.error("【订单消费】处理异常: messageId={}, orderNo={}", messageId, orderNo, e);
            
            // 更新消费失败状态
            consumeRecordService.updateConsumeStatus(messageId, "order_service_group", 
                ConsumeStatusEnum.FAILED, e.getMessage());
            
            // 抛出异常，RocketMQ会重试
            throw new RuntimeException("消费失败", e);
        }
    }
}
```

```java
@Configuration
public class RocketMQConfig {
    
    @Value("${rocketmq.name-server}")
    private String nameServer;
    
    @Bean
    public RocketMQTemplate rocketMQTemplate() {
        RocketMQTemplate template = new RocketMQTemplate();
        
        DefaultMQProducer producer = new DefaultMQProducer("payment-producer-group");
        producer.setNamesrvAddr(nameServer);
        producer.setSendMsgTimeout(3000);
        producer.setRetryTimesWhenSendFailed(2);
        producer.setRetryTimesWhenSendAsyncFailed(2);
        
        template.setProducer(producer);
        return template;
    }
    
    @Bean
    public DefaultMQPushConsumer orderPaymentConsumer(
            @Value("${rocketmq.consumer.order-service-group.topic}") String topic,
            @Value("${rocketmq.consumer.order-service-group.tag}") String tag,
            @Value("${rocketmq.consumer.order-service-group.consume-thread-max}") int consumeThreadMax) {
        
        DefaultMQPushConsumer consumer = new DefaultMQPushConsumer("order_service_group");
        consumer.setNamesrvAddr(nameServer);
        consumer.subscribe(topic, tag);
        consumer.setConsumeThreadMin(5);
        consumer.setConsumeThreadMax(consumeThreadMax);
        consumer.setConsumeMessageBatchMaxSize(1);  // 顺序消息必须设置为1
        consumer.registerMessageListener((MessageListenerOrderly) (msgs, context) -> {
            // 消息监听逻辑
            return ConsumeOrderlyStatus.SUCCESS;
        });
        
        return consumer;
    }
}
```

```java
    private DefaultMQAdminExt mqAdminExt;
    
    @Autowired
    private MetricsService metricsService;
    
    @Scheduled(fixedDelay = 60000)  // 每分钟检查一次
    public void monitorMQStatus() {
        try {
            // 1. 检查消息积压
            checkMessageBacklog();
            
            // 2. 检查消费延迟
            checkConsumeDelay();
            
            // 3. 检查死信队列
            checkDeadLetterQueue();
            
        } catch (Exception e) {
            log.error("监控MQ状态异常", e);
        }
    }
    
    private void checkMessageBacklog() throws Exception {
        String[] topics = {"payment_success_topic", "payment_failed_topic"};
        
        for (String topic : topics) {
            TopicStatsTable stats = mqAdminExt.examineTopicStats(topic);
            
            long totalBacklog = 0;
            for (Map.Entry<MessageQueue, TopicOffset> entry : stats.getOffsetTable().entrySet()) {
                long diff = entry.getValue().getMaxOffset() - entry.getValue().getMinOffset();
                totalBacklog += diff;
            }
            
            // 记录指标
            metricsService.recordGauge("mq.backlog", totalBacklog, "topic", topic);
            
            // 告警
            if (totalBacklog > 1000) {
                alertService.sendAlert("MQ积压告警", 
                    String.format("topic: %s, 积压消息数: %d", topic, totalBacklog));
            }
        }
    }
}
```

```java
@Component
@Slf4j
public class PaymentFailureHandler {
    
    /**
     * 处理RocketMQ不可用
     */
    public void handleMQUnavailable(PayOrder payOrder, Map<String, String> params) {
        try {
            // 1. 记录到数据库
            saveToFallbackTable(payOrder, params);
            
            // 2. 定时任务补偿
            scheduleCompensation(payOrder);
            
            // 3. 发送告警
            alertService.sendAlert("MQ不可用", 
                String.format("支付单号: %s, 已降级到数据库存储", payOrder.getPaymentNo()));
                
        } catch (Exception e) {
            log.error("处理MQ不可用异常", e);
        }
    }
    
    /**
     * 降级到数据库的表
     */
    @Table("pay_order_fallback")
    @Data
    public class PayOrderFallback {
        private String paymentNo;
        private String orderNo;
        private String params;
        private Integer status;
        private Date createTime;
    }
}
```
